Calculate interface pLDDT scores from AlphaFold-Multimer predicted nanobody-antigen complexes.

This script processes PDB files containing nanobody-antigen complexes predicted by
AlphaFold-Multimer and generates a CSV file with interface pLDDT confidence metrics
for each complex. Per-residue pLDDT scores are extracted from the B-factor column
of ATOM records (standard AlphaFold output format).

Interpretation of pLDDT scores (from AlphaFold2 paper):
  - pLDDT > 90: Very high confidence (expected RMSD < 1 Angstrom)
  - pLDDT > 70: High confidence (likely correct backbone prediction)
  - pLDDT > 50: Medium confidence (some regions may be incorrect)
  - pLDDT < 50: Low confidence (likely incorrect or disordered)

For interface regions specifically:
  - Mean interface pLDDT > 70 suggests a confidently predicted binding mode
  - Mean interface pLDDT < 50 suggests the predicted interface is unreliable

Dependencies: numpy, biopython
"""

def _extract_ca_data(chain: Chain, chain_id: str) -> Tuple[Dict[int, np.ndarray], Dict[int, float]]:
    """
    Extract CA coordinates and pLDDT scores for all residues in a chain.
    
    pLDDT scores are stored in the B-factor column of ATOM records in
    AlphaFold-Multimer output PDB files.
    
    Args:
        chain: Biopython Chain object
        chain_id: Chain identifier (for error messages)
        
    Returns:
        Tuple of (residue_number -> CA_coordinates dict, 
                  residue_number -> pLDDT_score dict)
    """
    coords: Dict[int, np.ndarray] = {}
    plddt: Dict[int, float] = {}
    for residue in chain:
        res_id = residue.get_id()[1]
        try:
            ca = residue["CA"]
            coords[res_id] = ca.get_vector().get_array()
            bfactor = ca.get_bfactor()
            # Validate and clamp pLDDT to expected range [0, 100]
            if not (0 <= bfactor <= 100):
                warnings.warn(
                    f"pLDDT value {bfactor:.1f} outside expected range [0,100] "
                    f"for residue {res_id} chain {chain_id}. Clamping."
                )
                bfactor = max(0.0, min(100.0, bfactor))
            plddt[res_id] = bfactor
        except KeyError:
            continue  # Skip residues without CA atoms (rare)
    return coords, plddt


def detect_chains_by_length(structure: Structure.Structure) -> Tuple[str, str]:
    """
    Assign nanobody (shorter) and antigen (longer) chains by residue count.
    
    Assumes the complex contains exactly two chains. The shorter chain is
    identified as the nanobody (typically ~120-150 residues) and the longer
    as the antigen (e.g., spike RBD ~200 residues).
    
    Args:
        structure: Biopython Structure object (first model used)
        
    Returns:
        Tuple of (nanobody_chain_id, antigen_chain_id)
        
    Raises:
        ValueError: If structure doesn't have exactly 2 chains
    """
    model = structure[0]
    chains = list(model.get_chains())
    
    if len(chains) != 2:
        raise ValueError(
            f"Auto-detection requires exactly 2 chains, found {len(chains)}. "
            f"Chains present: {[c.get_id() for c in chains]}"
        )
    
    chain_lengths = {
        chain.get_id(): sum(1 for _ in chain.get_residues())
        for chain in chains
    }
    
    sorted_chains = sorted(chain_lengths.items(), key=lambda x: x[1])
    return sorted_chains[0][0], sorted_chains[1][0]


def identify_interface_residues(
    model: Model.Model,
    chain1_id: str,
    chain2_id: str,
    distance_cutoff: float
) -> Tuple[List[int], List[int]]:
    """
    Identify interface residues between two chains using CA-CA distances.
    
    A residue is considered at the interface if its CA atom is within
    distance_cutoff of any CA atom in the opposing chain. This CA-only
    approach is standard for defining residue-level contacts in protein
    complexes, as CA positions are the most reliable atoms in predicted
    structures.
    
    Args:
        model: Biopython Model object (single model from a structure)
        chain1_id: First chain identifier
        chain2_id: Second chain identifier
        distance_cutoff: CA-CA distance cutoff in Angstroms
        
    Returns:
        Tuple of (interface residue numbers in chain1, interface residue numbers in chain2)
        
    Raises:
        KeyError: If either chain is not found in the model
    """
    if chain1_id not in model:
        raise KeyError(f"Chain '{chain1_id}' not found in model")
    if chain2_id not in model:
        raise KeyError(f"Chain '{chain2_id}' not found in model")
    
    # Extract CA coordinates for both chains
    coords1, _ = _extract_ca_data(model[chain1_id], chain1_id)
    coords2, _ = _extract_ca_data(model[chain2_id], chain2_id)
    
    if not coords1 or not coords2:
        warnings.warn(
            f"No CA atoms found in chain "
            f"{chain1_id if not coords1 else chain2_id}"
        )
        return [], []
    
    # Convert to arrays for vectorized distance computation
    residues1 = np.array(list(coords1.keys()))
    residues2 = np.array(list(coords2.keys()))
    positions1 = np.array(list(coords1.values()))
    positions2 = np.array(list(coords2.values()))
    
    # Compute all-pairs CA-CA distances using broadcasting
    diff = positions1[:, np.newaxis, :] - positions2[np.newaxis, :, :]
    distances = np.sqrt(np.sum(diff ** 2, axis=-1))
    
    # Find interface residues (any CA-CA pair within cutoff)
    contact_pairs = np.where(distances <= distance_cutoff)
    interface1 = sorted(set(residues1[contact_pairs[0]]))
    interface2 = sorted(set(residues2[contact_pairs[1]]))
    
    return interface1, interface2


def compute_interface_metrics(
    model: Model.Model,
    chain1_id: str,
    chain2_id: str,
    distance_cutoff: float,
    pdb_filename: str
) -> Dict[str, Any]:
    """
    Compute interface pLDDT metrics for a nanobody-antigen complex model.
    
    Calculates the mean pLDDT across all interface residues, where interface
    residues are defined by CA-CA distance. Provides both raw counts and
    normalized metrics for cross-complex comparison.
    
    Args:
        model: Biopython Model object containing the complex
        chain1_id: Chain ID for nanobody
        chain2_id: Chain ID for antigen
        distance_cutoff: CA-CA distance cutoff for interface definition
        pdb_filename: Original PDB filename (for warnings)
        
    Returns:
        Dictionary with keys:
            - mean_interface_plddt: mean pLDDT of all interface residues
            - mean_nanobody_interface_plddt: mean pLDDT of nanobody interface residues
            - mean_antigen_interface_plddt: mean pLDDT of antigen interface residues
            - num_nanobody_interface_residues: count of interface residues on nanobody
            - num_antigen_interface_residues: count of interface residues on antigen
            - total_interface_residues: sum of both interface residue counts
            - interface_residues_per_100_residues: normalized interface size
            - high_confidence_fraction: fraction of interface residues with pLDDT > 70
            - nanobody_chain: chain ID used for nanobody
            - antigen_chain: chain ID used for antigen
    """
    # Extract pLDDT scores
    _, plddt1 = _extract_ca_data(model[chain1_id], chain1_id)
    _, plddt2 = _extract_ca_data(model[chain2_id], chain2_id)
    
    # Identify interface residues
    interface1, interface2 = identify_interface_residues(
        model, chain1_id, chain2_id, distance_cutoff
    )
    
    # Collect interface pLDDT scores
    interface_plddt_1 = [plddt1[res] for res in interface1 if res in plddt1]
    interface_plddt_2 = [plddt2[res] for res in interface2 if res in plddt2]
    all_interface_plddt = interface_plddt_1 + interface_plddt_2
    
    # Compute total residues for normalization
    total_residues = len(plddt1) + len(plddt2)
    
    # Compute metrics with proper handling of empty interface
    if all_interface_plddt:
        mean_interface = float(np.mean(all_interface_plddt))
        mean_chain1 = float(np.mean(interface_plddt_1)) if interface_plddt_1 else float('nan')
        mean_chain2 = float(np.mean(interface_plddt_2)) if interface_plddt_2 else float('nan')
        high_conf = sum(1 for p in all_interface_plddt if p > 70) / len(all_interface_plddt)
        interface_per_100 = (len(interface1) + len(interface2)) / total_residues * 100 if total_residues > 0 else float('nan')
    else:
        mean_interface = float('nan')
        mean_chain1 = float('nan')
        mean_chain2 = float('nan')
        high_conf = float('nan')
        interface_per_100 = 0.0
        warnings.warn(
            f"No interface residues found in {pdb_filename} "
            f"(cutoff={distance_cutoff}A). Try increasing distance cutoff."
        )
    
    return {
        "mean_interface_plddt": mean_interface,
        "mean_nanobody_interface_plddt": mean_chain1,
        "mean_antigen_interface_plddt": mean_chain2,
        "num_nanobody_interface_residues": len(interface1),
        "num_antigen_interface_residues": len(interface2),
        "total_interface_residues": len(interface1) + len(interface2),
        "interface_residues_per_100_residues": interface_per_100,
        "high_confidence_fraction": high_conf,
        "nanobody_chain": chain1_id,
        "antigen_chain": chain2_id
    }

def write_csv(results: List[Dict[str, Any]], output_path: str) -> None:
    """
    Write results to CSV file with consistent formatting.
    
    NaN values are written as empty strings for easier import into
    spreadsheet software and analysis tools.
    
    Args:
        results: List of result dictionaries
        output_path: Path to output CSV file
    """
    if not results:
        print("No results to write.", file=sys.stderr)
        return
    
    fieldnames = [
        "complex_name",
        "pdb_file",
        "nanobody_chain",
        "antigen_chain",
        "mean_interface_plddt",
        "mean_nanobody_interface_plddt",
        "mean_antigen_interface_plddt",
        "num_nanobody_interface_residues",
        "num_antigen_interface_residues",
        "total_interface_residues",
        "interface_residues_per_100_residues",
        "high_confidence_fraction"
    ]
    
    if any("error" in entry for entry in results):
        fieldnames.append("error")
    
    with open(output_path, 'w', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames, extrasaction='ignore')
        writer.writeheader()
        
        for entry in results:
            # Format NaN values as empty strings for cleaner output
            row = {}
            for key, value in entry.items():
                if isinstance(value, float) and np.isnan(value):
                    row[key] = ""
                else:
                    row[key] = value
            writer.writerow(row)
    
    print(f"Results written to {output_path}", file=sys.stderr)

def validate_plddt_scores(plddt_scores: Dict) -> None:
    """
    Validate that pLDDT scores are within the expected range [0, 100].
    
    AlphaFold-Multimer outputs pLDDT scores in the range 0-100, where higher
    values indicate higher confidence. This function checks for values outside
    this range and warns if anomalies are detected.
    
    Args:
        plddt_scores: Dictionary of residue ID to pLDDT score
        
    Raises:
        ValueError: If pLDDT values are outside the valid range
    """
    if not plddt_scores:
        return
    
    all_scores = list(plddt_scores.values())
    max_score = max(all_scores)
    min_score = min(all_scores)
    
    if max_score > 100 or min_score < 0:
        raise ValueError(
            f"pLDDT scores outside expected range [0, 100]: "
            f"min={min_score:.2f}, max={max_score:.2f}. "
            f"This may indicate corrupted PDB files or incorrect parsing."
        )


def get_plddt_from_model(model: Model, chain_id: str) -> Dict[str, float]:
    """
    Extract pLDDT scores from B-factor column for a specific chain in a single model.
    
    AlphaFold-Multimer stores per-residue pLDDT confidence scores in the B-factor
    field of ATOM records. Since pLDDT is a per-residue metric, all atoms in a
    residue have identical B-factors; we take the first atom's value.
    
    Args:
        model: Bio.PDB Model object
        chain_id: Chain identifier to extract scores from
        
    Returns:
        Dictionary mapping residue identifiers (resnum, insertion code) to pLDDT score
        
    Raises:
        ValueError: If no residues are found for the specified chain
    """
    plddt_scores = {}
    
    for chain in model:
        if chain.get_id() != chain_id:
            continue
        
        for residue in chain:
            # Skip heteroatoms and water (residue IDs with non-space first char)
            if residue.get_id()[0] != ' ':
                continue
            
            # pLDDT is per-residue; take the B-factor from the first atom
            atoms = list(residue.get_atoms())
            if atoms:
                residue_id = residue.get_id()
                plddt_scores[residue_id] = atoms[0].get_bfactor()
    
    if not plddt_scores:
        raise ValueError(f"No residues found for chain {chain_id} in model")
    
    validate_plddt_scores(plddt_scores)
    return plddt_scores


def get_interface_residues_from_model(
    model: Model,
    chain_nanobody: str,
    chain_antigen: str,
    distance_cutoff: float,
    min_atom_contacts: int = 1
) -> List[str]:
    """
    Identify nanobody interface residues in a single model.
    
    Uses Bio.PDB.NeighborSearch for efficient spatial querying. A nanobody residue
    is considered an interface residue if it has at least `min_atom_contacts` atoms
    within `distance_cutoff` Å of any antigen atom.
    
    Args:
        model: Bio.PDB Model object
        chain_nanobody: Chain ID for nanobody
        chain_antigen: Chain ID for antigen
        distance_cutoff: Distance in Angstroms for defining contact
        min_atom_contacts: Minimum number of atom-atom contacts to define interface
        
    Returns:
        List of nanobody residue identifiers (residue_id tuples) at the interface
    """
    # Collect atoms for each chain
    nb_atoms = []
    ag_atom_objects = []
    
    for chain in model:
        chain_id = chain.get_id()
        if chain_id == chain_nanobody:
            for residue in chain:
                if residue.get_id()[0] != ' ':
                    continue
                for atom in residue:
                    nb_atoms.append((atom, residue.get_id()))
        elif chain_id == chain_antigen:
            for residue in chain:
                if residue.get_id()[0] != ' ':
                    continue
                for atom in residue:
                    ag_atom_objects.append(atom)
    
    if not ag_atom_objects:
        return []
    
    # Build neighbor search on antigen atoms for efficient spatial queries
    ns = NeighborSearch(ag_atom_objects)
    
    # For each nanobody atom, find antigen atoms within cutoff
    nb_contact_count = {}  # residue_id -> number of atom contacts
    
    for nb_atom, nb_resid in nb_atoms:
        # Query for nearby antigen atoms
        nearby = ns.search(nb_atom.get_vector(), distance_cutoff)
        if nearby:
            nb_contact_count[nb_resid] = nb_contact_count.get(nb_resid, 0) + len(nearby)
    
    # Filter nanobody residues by minimum contact count
    nb_interface = [
        resid for resid, count in nb_contact_count.items()
        if count >= min_atom_contacts
    ]
    
    return list(nb_interface)


def compute_interface_plddt_for_model(
    model: Model,
    chain_nanobody: str,
    chain_antigen: str,
    distance_cutoff: float,
    min_atom_contacts: int
) -> Tuple[Optional[float], int, int]:
    """
    Compute interface pLDDT for a single model.
    
    Returns None for interface_plddt if the interface cannot be computed
    (e.g., missing chain, pLDDT validation failure).
    
    Args:
        model: Bio.PDB Model object
        chain_nanobody: Chain ID for nanobody
        chain_antigen: Chain ID for antigen
        distance_cutoff: Distance cutoff in Angstroms
        min_atom_contacts: Minimum atom contacts for interface definition
        
    Returns:
        Tuple of (interface_plddt, num_interface_residues, total_nanobody_residues)
        where interface_plddt is None if no interface found or error occurred
    """
    try:
        nb_plddt = get_plddt_from_model(model, chain_nanobody)
    except ValueError:
        return None, 0, 0
    
    # Identify interface residues
    nb_interface_res = get_interface_residues_from_model(
        model, chain_nanobody, chain_antigen,
        distance_cutoff, min_atom_contacts
    )
    
    if not nb_interface_res:
        return None, 0, len(nb_plddt)
    
    # Calculate average pLDDT of interface residues
    interface_scores = [
        nb_plddt[res] for res in nb_interface_res if res in nb_plddt
    ]
    
    if not interface_scores:
        return None, len(nb_interface_res), len(nb_plddt)
    
    interface_plddt = np.mean(interface_scores)
    
    return interface_plddt, len(nb_interface_res), len(nb_plddt)


def compute_interface_plddt(
    pdb_file: str,
    chain_nanobody: str,
    chain_antigen: str,
    distance_cutoff: float,
    min_atom_contacts: int,
    model_index: int
) -> Tuple[str, Optional[float], int, int, str]:
    """
    Compute the interface pLDDT for a single nanobody-antigen complex.
    
    The interface pLDDT is defined as the mean pLDDT of nanobody residues that
    have at least `min_atom_contacts` atom-atom contacts within `distance_cutoff` Å
    of any antigen residue.
    
    Args:
        pdb_file: Path to PDB file (may be .pdb or .pdb.gz)
        chain_nanobody: Chain ID for nanobody
        chain_antigen: Chain ID for antigen
        distance_cutoff: Distance cutoff in Angstroms
        min_atom_contacts: Minimum atom contacts for interface definition
        model_index: Model index to use (-1 for average across all models)
        
    Returns:
        Tuple of (filename, interface_plddt, num_interface_residues, 
                  total_nanobody_residues, status_message)
        where interface_plddt is None if no interface detected or error occurred
    """
    parser = PDBParser(QUIET=True)
    
    # Handle both plain and gzipped PDB files
    try:
        if pdb_file.endswith('.gz'):
            with gzip.open(pdb_file, 'rt') as f:
                structure = parser.get_structure("complex", f)
        else:
            structure = parser.get_structure("complex", pdb_file)
    except Exception as e:
        return (os.path.basename(pdb_file), None, 0, 0, f"Parse error: {str(e)}")
    
    # Validate that specified chains exist
    chain_ids = set()
    for model in structure:
        for chain in model:
            chain_ids.add(chain.get_id())
    
    if chain_nanobody not in chain_ids:
        return (os.path.basename(pdb_file), None, 0, 0,
                f"Nanobody chain '{chain_nanobody}' not found. Available chains: {sorted(chain_ids)}")
    
    if chain_antigen not in chain_ids:
        return (os.path.basename(pdb_file), None, 0, 0,
                f"Antigen chain '{chain_antigen}' not found. Available chains: {sorted(chain_ids)}")
    
    # Get list of models
    model_list = list(structure.get_models())
    
    # Validate model index (only for non-average mode)
    if model_index != -1 and model_index >= len(model_list):
        return (os.path.basename(pdb_file), None, 0, 0,
                f"Model index {model_index} out of range (0-{len(model_list)-1})")
    
    if model_index == -1:
        # Average interface pLDDT across all models.
        # pLDDT scores are identical across models (per-position confidence),
        # but interface residues may differ due to different conformations.
        all_interface_plddts = []
        total_interface_res = 0
        total_nb_res = 0
        
        for model in model_list:
            iplddt, n_int, n_tot = compute_interface_plddt_for_model(
                model, chain_nanobody, chain_antigen,
                distance_cutoff, min_atom_contacts
            )
            if iplddt is not None:
                all_interface_plddts.append(iplddt)
                total_interface_res += n_int
                total_nb_res += n_tot
        
        if all_interface_plddts:
            avg_plddt = np.mean(all_interface_plddts)
            avg_interface_res = total_interface_res // len(model_list)
            avg_nb_res = total_nb_res // len(model_list)
            return (os.path.basename(pdb_file), avg_plddt,
                    avg_interface_res, avg_nb_res, f"Average of {len(all_interface_plddts)} models")
        else:
            return (os.path.basename(pdb_file), None, 0, 0,
                    "No interface found in any model")
    else:
        # Use specific model
        selected_model = model_list[model_index]
        iplddt, n_int, n_tot = compute_interface_plddt_for_model(
            selected_model, chain_nanobody, chain_antigen,
            distance_cutoff, min_atom_contacts
        )
        
        if iplddt is None:
            return (os.path.basename(pdb_file), None, n_int, n_tot,
                    "No interface residues detected")
        
        return (os.path.basename(pdb_file), iplddt, n_int, n_tot, "OK")


def process_directory(
    input_dir: str,
    chain_nanobody: str,
    chain_antigen: str,
    distance_cutoff: float,
    min_atom_contacts: int,
    model_index: int
) -> List[Tuple[str, Optional[float], int, int, str]]:
    """
    Process all PDB files in a directory.
    
    Args:
        input_dir: Directory containing PDB files
        chain_nanobody: Chain ID for nanobody
        chain_antigen: Chain ID for antigen
        distance_cutoff: Distance cutoff for interface definition
        min_atom_contacts: Minimum atom contacts for interface definition
        model_index: Model index to use (-1 for average)
        
    Returns:
        List of tuples with results for each PDB file
    """
    # Find all PDB files (both .pdb and .pdb.gz)
    pdb_files = glob.glob(os.path.join(input_dir, "*.pdb"))
    pdb_files.extend(glob.glob(os.path.join(input_dir, "*.pdb.gz")))
    
    if not pdb_files:
        print(f"Warning: No PDB files found in {input_dir}")
        return []
    
    results = []
    for pdb_file in sorted(pdb_files):
        result = compute_interface_plddt(
            pdb_file, chain_nanobody, chain_antigen,
            distance_cutoff, min_atom_contacts, model_index
        )
        results.append(result)
        
        # Print progress with appropriate formatting
        fname, iplddt, n_int, n_tot, status = result
        if iplddt is not None:
            print(f"✓ {fname}: interface pLDDT = {iplddt:.2f} "
                  f"({n_int}/{n_tot} residues) [{status}]")
        else:
            print(f"⚠ {fname}: No interface detected ({status})")
    
    return results


def write_output(
    results: List[Tuple[str, Optional[float], int, int, str]],
    output_file: str
) -> None:
    """
    Write results to CSV file.

    - filename: Name of the PDB file
    - interface_pLDDT: Mean pLDDT of interface residues (higher = better confidence).
                       'N/A' indicates no interface was detected.

    - total_nanobody_residues: Total number of nanobody residues
    - interface_fraction: Proportion of nanobody at interface (typically 10-25%).

    - status: Processing status message

        results: List of (filename, interface_plddt, num_interface, 
                total_nanobody, status) tuples
        output_file: Path to output CSV file
    """
    with open(output_file, 'w', newline='') as csvfile:
        writer = csv.writer(csvfile)
        writer.writerow([
            'filename',
            'interface_pLDDT',
            'num_interface_residues',
            'total_nanobody_residues',
            'interface_fraction',
            'status'
        ])
        
        for filename, interface_plddt, num_interface, total_nanobody, status in results:
            if interface_plddt is not None:
                interface_fraction = num_interface / total_nanobody if total_nanobody > 0 else 0
                writer.writerow([

                    total_nanobody,

                writer.writerow([
                    filename,
                    'N/A',
                    num_interface,
                    total_nanobody,
                    'N/A',

                ])
    
    print(f"\nResults written to {output_file}")

#!/usr/bin/env python3
"""
Process AlphaFold-Multimer nanobody-antigen complex predictions and compute
interface pLDDT (predicted Local Distance Difference Test) scores.

This script takes a directory of PDB files (each containing one nanobody-antigen
complex) and outputs a CSV with complex names, interface pLDDT statistics, and
number of interface residues detected.

Interface pLDDT is defined as the average pLDDT score of all nanobody and antigen
residues that have at least one atom within the specified distance cutoff.

Higher interface pLDDT values indicate more confident interface predictions.
These are empirical guidelines, not guarantees of experimental binding:
  - > 90: Very high confidence (predicted binding mode is likely correct)
  - 70-90: High confidence (strong candidates for experimental validation)
  - 50-70: Medium confidence (may bind but prediction less reliable)
  - < 50: Low confidence (unlikely to represent true binding interface)

CHAIN ASSIGNMENT HEURISTIC:
By default, the script assumes the smallest chain(s) by residue count are the
nanobody and the largest chain(s) are the antigen. This works for typical
monomeric nanobody (~120 residues) vs spike monomer (~600+ residues).
For multimeric constructs (e.g., dimeric nanobody, trimeric spike), use
--nanobody_chains and --antigen_chains to specify all relevant chain IDs.

    python process_af2_complexes.py --input_dir /path/to/pdbs --output results.csv
    
    # Specify multiple chains for multimeric complexes:

def infer_chains_by_size(structure: Structure, model_index: int = 0) -> Tuple[List[str], List[str]]:
    """
    Infer nanobody and antigen chain assignments based on residue counts.
    
    Heuristic: Chains are sorted by residue count. The smallest chain(s)
    are assigned as nanobody (assuming ~120 residues for a single nanobody
    domain), and the largest chain(s) are assigned as antigen (assuming
    ~600+ residues for a spike protein monomer). For multimeric complexes,
    this heuristic may fail and users should specify chains explicitly.
    
    The heuristic assigns chains as follows:
    - If 2 chains: smallest = nanobody, largest = antigen
    - If 3 chains: smallest = nanobody, two largest = antigen (e.g., nanobody + spike dimer)
    - If 4+ chains: smallest two = nanobody, remaining = antigen (e.g., dimeric nanobody + trimeric spike)
    
    Args:
        structure: Biopython Structure object
        model_index: Which model to use (0 = ranked_0)
        
    Returns:
        Tuple of (nanobody_chain_list, antigen_chain_list)
        
    Raises:
        ValueError: If fewer than 2 chains with standard residues are found
    """
    models = list(structure.get_models())
    if model_index >= len(models):
        raise ValueError(
            f"Model index {model_index} not found (only {len(models)} models available)"
        )
    model = models[model_index]
    
    chain_residue_counts: Dict[str, int] = {}
    for chain in model:
        chain_id = chain.get_id()
        residue_count = sum(
            1 for res in chain if res.get_id()[0] == " "
        )
        if residue_count > 0:
            chain_residue_counts[chain_id] = residue_count
    
    if len(chain_residue_counts) < 2:
        raise ValueError(
            f"Expected at least 2 chains with standard residues, "
            f"found {len(chain_residue_counts)}: {list(chain_residue_counts.keys())}. "
            f"Use --nanobody_chains and --antigen_chains to specify manually."
        )
    
    # Sort chains by residue count (ascending)
    sorted_chains = sorted(chain_residue_counts.items(), key=lambda x: x[1])
    
    # Assign chains: smallest 1-2 chains as nanobody, rest as antigen
    num_chains = len(sorted_chains)
    if num_chains == 2:
        nb_chains = [sorted_chains[0][0]]
        ag_chains = [sorted_chains[1][0]]
    elif num_chains == 3:
        nb_chains = [sorted_chains[0][0]]
        ag_chains = [sorted_chains[1][0], sorted_chains[2][0]]
    else:  # 4+ chains
        nb_chains = [sorted_chains[0][0], sorted_chains[1][0]]
        ag_chains = [c[0] for c in sorted_chains[2:]]
    
    nb_residues = sum(chain_residue_counts[c] for c in nb_chains)
    ag_residues = sum(chain_residue_counts[c] for c in ag_chains)
    
    print(f"  Inferred: nanobody chains {nb_chains} "
          f"(total {nb_residues} residues), "
          f"antigen chains {ag_chains} "
          f"(total {ag_residues} residues)")
    
    return nb_chains, ag_chains


def compute_interface_plddt(
    structure: Structure,
    nanobody_chains: List[str],
    antigen_chains: List[str],
    cutoff: float,
    model_index: int = 0
) -> Optional[Tuple[float, float, float, float, int]]:
    """
    Compute interface pLDDT statistics for a nanobody-antigen complex.
    
    Extracts pLDDT scores from the B-factor column of ATOM records (standard
    AlphaFold2/Multimer output, where pLDDT is stored as B-factors on a 0-100
    scale). Identifies interface residues as any residue with at least one
    atom within `cutoff` Angstroms of the opposite chain(s).
    
    Args:
        structure: Parsed Biopython Structure object
        nanobody_chains: List of chain IDs for the nanobody
        antigen_chains: List of chain IDs for the antigen
        cutoff: Distance cutoff for defining interface residues (Angstroms)
        model_index: Which model to use (0 = ranked_0, the best model)
        
    Returns:
        Tuple of (mean_pLDDT, std_pLDDT, min_pLDDT, max_pLDDT, num_residues),
        or None if processing fails or no interface is detected
    """
    try:
        models = list(structure.get_models())
        if model_index >= len(models):
            print(f"  Warning: Model index {model_index} not found "
                  f"(only {len(models)} models available). "
                  f"Available indices: 0-{len(models)-1}", file=sys.stderr)
            return None
        model = models[model_index]
        
        # Build sets for fast chain membership lookup
        nb_chain_set = set(nanobody_chains)
        ag_chain_set = set(antigen_chains)
        
        # Collect atoms and residue-level pLDDT for each chain group
        nb_atoms = []
        ag_atoms = []
        residue_plddt: Dict[Tuple[str, int], float] = {}
        
        for chain in model:
            chain_id = chain.get_id()
            is_nb = chain_id in nb_chain_set
            is_ag = chain_id in ag_chain_set
            
            if not is_nb and not is_ag:
                continue  # Skip chains not assigned to either group
            
            for residue in chain:
                if residue.get_id()[0] != " ":
                    continue  # Skip HETATM and water
                
                res_id = residue.get_id()[1]
                atoms = list(residue.get_atoms())
                if not atoms:
                    continue
                
                # pLDDT is stored in B-factor column (0-100 scale)
                plddt = atoms[0].get_bfactor()
                residue_plddt[(chain_id, res_id)] = plddt
                
                if is_nb:
                    nb_atoms.extend(atoms)
                else:
                    ag_atoms.extend(atoms)
        
        # Validate we have atoms from both groups
        if not nb_atoms:
            print(f"  Warning: No atoms found for nanobody chains {nanobody_chains}",
                  file=sys.stderr)
            return None
        if not ag_atoms:
            print(f"  Warning: No atoms found for antigen chains {antigen_chains}",
                  file=sys.stderr)
            return None
        
        # Validate pLDDT scores are in expected range (0-100)
        all_plddt = list(residue_plddt.values())
        if all_plddt:
            min_all, max_all = min(all_plddt), max(all_plddt)
            if max_all > 100 or min_all < 0:
                print(f"  Warning: pLDDT scores out of range [0,100] "
                      f"(min={min_all:.1f}, max={max_all:.1f}). "
                      f"Are these AlphaFold PDB files?", file=sys.stderr)
        
        # Find interface residues using neighbor search
        # Combine all atoms from both groups into a single search structure
        all_atoms = nb_atoms + ag_atoms
        ns = NeighborSearch(all_atoms)
        
        # Find all atom pairs within cutoff
        contact_pairs = ns.search_all(cutoff, level="A")
        
        # Collect unique interface residue identifiers
        interface_residues: set = set()
        for atom1, atom2 in contact_pairs:
            res1 = atom1.get_parent()
            res2 = atom2.get_parent()
            
            chain1 = res1.get_parent().get_id()
            chain2 = res2.get_parent().get_id()
            res_id1 = res1.get_id()[1]
            res_id2 = res2.get_id()[1]
            
            # Include residues from both groups when they contact each other
            # (Skip same-group contacts - we want cross-group interface)
            in_nb1 = chain1 in nb_chain_set
            in_ag1 = chain1 in ag_chain_set
            in_nb2 = chain2 in nb_chain_set
            in_ag2 = chain2 in ag_chain_set
            
            if (in_nb1 and in_ag2) or (in_ag1 and in_nb2):
                interface_residues.add((chain1, res_id1))
                interface_residues.add((chain2, res_id2))
        
        if not interface_residues:
            print(f"  Warning: No interface residues detected within {cutoff}Å "
                  f"between nanobody {nanobody_chains} and antigen {antigen_chains}")
            return None
        
        # Compute pLDDT statistics for interface residues
        interface_plddt_values = [
            residue_plddt[res_id] for res_id in interface_residues
            if res_id in residue_plddt
        ]
        
        if not interface_plddt_values:
            return None
        
        mean_plddt = float(np.mean(interface_plddt_values))
        std_plddt = float(np.std(interface_plddt_values))
        min_plddt = float(np.min(interface_plddt_values))
        max_plddt = float(np.max(interface_plddt_values))
        num_residues = len(interface_residues)
        
        return (mean_plddt, std_plddt, min_plddt, max_plddt, num_residues)
    
    except Exception as e:
        print(f"  Error computing interface: {e}", file=sys.stderr)
        return None

#!/usr/bin/env python3
"""
Process AlphaFold-Multimer predicted nanobody-antigen complex PDB files
and calculate interface pLDDT confidence metrics.

This script reads PDB files from a specified directory, identifies the
interface between nanobody and antigen chains (user-specified), and computes
the average pLDDT (predicted Local Distance Difference Test) score for
interface residues. pLDDT is stored in the B-factor column of AlphaFold PDB
outputs (range 0-100, higher = more confident).

Interface pLDDT is calculated as the average per-residue pLDDT over all
interface residues. Each residue contributes its average pLDDT (averaged
over all atoms in that residue), giving equal weight to each interface
residue regardless of size.

IMPORTANT: pLDDT values are calibrated per-model in AlphaFold-Multimer.
For fair comparisons, only compare pLDDT values between complexes from
the same model rank (e.g., compare all model_1 complexes together).

Usage:
    python compute_interface_plddt.py \\
        --input_dir /path/to/pdb/files \\
        --output results.csv \\
        --nanobody_chain A \\
        --antigen_chain B
"""

def is_alphafold_pdb(pdb_filepath: str) -> Tuple[bool, str]:
    """
    Check if a PDB file appears to be from AlphaFold-Multimer.
    
    Looks for AlphaFold-specific REMARK records in the PDB header.
    This is a simple, reliable check without complex heuristics.
    
    Args:
        pdb_filepath: Path to the PDB file
        
    Returns:
        Tuple of (is_alphafold: bool, message: str)
    """
    try:
        with open(pdb_filepath, 'r') as f:
            for line in f:
                if line.startswith('REMARK'):
                    if 'ALPHAFOLD' in line.upper():
                        return True, "AlphaFold-Multimer identified via REMARK record"
                elif line.startswith('ATOM'):
                    # Reached atom records without finding AlphaFold remark
                    break
        return False, "No AlphaFold REMARK records found; may not be AlphaFold output"
    except Exception as e:
        return False, f"Error reading file: {str(e)}"


def parse_pdb_atom_lines(pdb_filepath: str) -> List[Dict[str, Any]]:
    """
    Parse ATOM records from a PDB file and extract relevant fields.
    
    Args:
        pdb_filepath: Path to the PDB file
        
    Returns:
        List of dictionaries containing atom information:
        - chain_id: Chain identifier (string)
        - res_id: Residue sequence number
        - res_name: Three-letter residue name
        - atom_name: Atom name
        - x, y, z: Cartesian coordinates
        - b_factor: B-factor (pLDDT for AlphaFold, range 0-100)
        - is_heavy: True if not a hydrogen atom
        
    Raises:
        FileNotFoundError: If the PDB file doesn't exist
        ValueError: If no ATOM records are found
    """
    atoms = []
    
    try:
        with open(pdb_filepath, 'r') as f:
            for line in f:
                if line.startswith('ATOM'):
                    try:
                        chain_id = line[21].strip() if len(line) > 21 else ''
                        if not chain_id:
                            continue
                        
                        atom_info = {
                            'chain_id': chain_id,
                            'res_id': int(line[22:26].strip()) if len(line) > 26 else 0,
                            'res_name': line[17:20].strip() if len(line) > 20 else '',
                            'atom_name': line[12:16].strip() if len(line) > 16 else '',
                            'x': float(line[30:38].strip()) if len(line) > 38 else 0.0,
                            'y': float(line[38:46].strip()) if len(line) > 46 else 0.0,
                            'z': float(line[46:54].strip()) if len(line) > 54 else 0.0,
                            'b_factor': float(line[60:66].strip()) if len(line) > 66 else 0.0,
                            'is_heavy': not line[12:16].strip().startswith('H') if len(line) > 16 else True
                        }
                        
                        atoms.append(atom_info)
                    except (ValueError, IndexError):
                        continue
    except FileNotFoundError:
        raise FileNotFoundError(f"PDB file not found: {pdb_filepath}")
    
    if not atoms:
        raise ValueError(f"No ATOM records found in {pdb_filepath}")
    
    return atoms


def compute_per_residue_plddt(atoms: List[Dict[str, Any]]) -> Dict[Tuple[str, int], float]:
    """
    Compute per-residue average pLDDT for all residues.
    
    Each residue's pLDDT is averaged over all its atoms (both heavy and
    hydrogen), giving equal weight to each residue regardless of size.
    
    Args:
        atoms: List of atom dictionaries
        
    Returns:
        Dictionary mapping (chain_id, res_id) to average pLDDT
    """
    plddt_sums: Dict[Tuple[str, int], float] = {}
    plddt_counts: Dict[Tuple[str, int], int] = {}
    
    for atom in atoms:
        key = (atom['chain_id'], atom['res_id'])
        if key not in plddt_sums:
            plddt_sums[key] = 0.0
            plddt_counts[key] = 0
        plddt_sums[key] += atom['b_factor']
        plddt_counts[key] += 1
    
    return {k: plddt_sums[k] / plddt_counts[k] for k in plddt_sums}


def compute_residue_centroids(atoms: List[Dict[str, Any]]) -> Dict[Tuple[str, int], np.ndarray]:
    """
    Compute centroid (average position) for each residue using all atoms.
    
    Args:
        atoms: List of atom dictionaries (assumed to be from a single chain)
        
    Returns:
        Dictionary mapping (chain_id, residue_id) tuples to centroid coordinates
    """
    residue_coords: Dict[Tuple[str, int], List[np.ndarray]] = {}
    
    for atom in atoms:
        key = (atom['chain_id'], atom['res_id'])
        if key not in residue_coords:
            residue_coords[key] = []
        residue_coords[key].append(np.array([atom['x'], atom['y'], atom['z']]))
    
    return {k: np.mean(coords, axis=0) for k, coords in residue_coords.items()}


def compute_interface_plddt(
    pdb_filepath: str,
    nanobody_chain: str,
    antigen_chain: str,
    distance_cutoff: float = 5.0
) -> Dict[str, Any]:
    """
    Compute interface pLDDT for a single AlphaFold-Multimer predicted complex.
    
    Interface residues are defined as those from the nanobody and antigen where
    the residue centroids (average of all atom positions) are within
    distance_cutoff of each other. pLDDT is then averaged over these interface
    residues, with each residue contributing its per-residue average pLDDT.
    
    Args:
        pdb_filepath: Path to the PDB file
        nanobody_chain: Chain ID for the nanobody
        antigen_chain: Chain ID for the antigen
        distance_cutoff: Distance cutoff in Angstroms for defining interface
                        contacts. Typical values:
                        - 4.0 Å: van der Waals contacts only
                        - 5.0 Å: standard side-chain interactions (default)
                        - 6.0 Å: includes solvent-mediated contacts
        
    Returns:
        Dictionary containing:
        - 'filename': Name of the PDB file
        - 'interface_plddt': Average per-residue pLDDT of interface residues
        - 'interface_residue_count': Total number of interface residues
        - 'nanobody_interface_count': Number of nanobody interface residues
        - 'antigen_interface_count': Number of antigen interface residues
        - 'high_confidence_interface_fraction': Fraction of interface residues
                                               with per-residue pLDDT > 70
        - 'is_alphafold': Whether file appears to be AlphaFold output
        - 'validation_message': Details about AlphaFold validation
        - 'error': Error message if processing failed, None otherwise
    """
    result: Dict[str, Any] = {
        'filename': os.path.basename(pdb_filepath),
        'interface_plddt': 0.0,
        'interface_residue_count': 0,
        'nanobody_interface_count': 0,
        'antigen_interface_count': 0,
        'high_confidence_interface_fraction': 0.0,
        'is_alphafold': False,
        'validation_message': '',
        'error': None
    }
    
    try:
        # Validate this is an AlphaFold output
        is_af, msg = is_alphafold_pdb(pdb_filepath)
        result['is_alphafold'] = is_af
        result['validation_message'] = msg
        
        if not is_af:
            print(f"  Warning: {result['filename']} may not be AlphaFold output - {msg}")
        
        # Parse atoms
        atoms = parse_pdb_atom_lines(pdb_filepath)
        
        # Validate chains exist
        available_chains = set(atom['chain_id'] for atom in atoms)
        if nanobody_chain not in available_chains:
            raise ValueError(f"Nanobody chain '{nanobody_chain}' not found. "
                           f"Available chains: {sorted(available_chains)}")
        if antigen_chain not in available_chains:
            raise ValueError(f"Antigen chain '{antigen_chain}' not found. "
                           f"Available chains: {sorted(available_chains)}")
        if nanobody_chain == antigen_chain:
            raise ValueError(f"Nanobody and antigen chains must be different")
        
        # Separate atoms by chain
        nb_atoms = [a for a in atoms if a['chain_id'] == nanobody_chain]
        ag_atoms = [a for a in atoms if a['chain_id'] == antigen_chain]
        
        if not nb_atoms:
            raise ValueError(f"No atoms found for nanobody chain '{nanobody_chain}'")
        if not ag_atoms:
            raise ValueError(f"No atoms found for antigen chain '{antigen_chain}'")
        
        # Compute residue centroids for efficient interface detection
        nb_centroids = compute_residue_centroids(nb_atoms)
        ag_centroids = compute_residue_centroids(ag_atoms)
        
        # Find interface residues using centroid distances
        nb_interface = set()
        ag_interface = set()
        
        for nb_key, nb_cent in nb_centroids.items():
            for ag_key, ag_cent in ag_centroids.items():
                dist = np.linalg.norm(nb_cent - ag_cent)
                if dist < distance_cutoff:
                    nb_interface.add(nb_key)
                    ag_interface.add(ag_key)
        
        # Validate interface size
        total_interface = len(nb_interface) + len(ag_interface)
        if total_interface < 5:
            print(f"  Warning: Very small interface ({total_interface} residues) "
                  f"in {result['filename']}; complex may not be valid")
        elif total_interface > 100:
            print(f"  Warning: Very large interface ({total_interface} residues) "
                  f"in {result['filename']}; may include non-specific contacts")
        
        # Compute per-residue pLDDT
        all_res_plddt = compute_per_residue_plddt(atoms)
        
        # Get pLDDT values for interface residues (one value per residue)
        nb_plddt_values = [all_res_plddt[k] for k in nb_interface if k in all_res_plddt]
        ag_plddt_values = [all_res_plddt[k] for k in ag_interface if k in all_res_plddt]
        all_plddt_values = nb_plddt_values + ag_plddt_values
        
        # Calculate metrics
        if all_plddt_values:
            result['interface_plddt'] = float(np.mean(all_plddt_values))
            high_conf = sum(1 for v in all_plddt_values if v > 70)
            result['high_confidence_interface_fraction'] = high_conf / len(all_plddt_values)
        
        result['interface_residue_count'] = total_interface
        result['nanobody_interface_count'] = len(nb_interface)
        result['antigen_interface_count'] = len(ag_interface)
        
    except Exception as e:
        result['error'] = str(e)
    
    return result

def make_residue_key(chain_id: str, residue) -> str:
    """
    Create a canonical residue identifier handling insertion codes.
    
    PDB residues can have insertion codes (e.g., residue 100A vs 100B).
    This function creates a unique string key that handles this.
    
    Args:
        chain_id: Chain identifier string
        residue: Biopython Residue object
        
    Returns:
        Canonical key like "A:100" or "A:100A" if insertion code present
    """
    res_id = residue.get_id()
    res_num = res_id[1]
    insertion_code = res_id[2]
    
    if insertion_code != " ":
        return f"{chain_id}:{res_num}{insertion_code}"
    else:
        return f"{chain_id}:{res_num}"


def extract_plddt_from_pdb(structure: Structure) -> Dict[str, float]:
    """
    Extract per-residue pLDDT scores from the B-factor column.
    
    AlphaFold-Multimer stores the predicted LDDT score (0-100) in the 
    B-factor column of all ATOM records. We use the CA atom's B-factor
    as the residue-level pLDDT.
    
    Args:
        structure: Biopython Structure object
        
    Returns:
        Dictionary mapping residue identifier (chain:residue_number) to pLDDT score
        
    Raises:
        ValueError: If no valid pLDDT scores are found or scores are out of range
    """
    plddt_scores = {}
    
    for model in structure:
        for chain in model:
            for residue in chain:
                # Skip heteroatoms (water, ligands)
                if residue.get_id()[0] != " ":
                    continue
                
                # Create canonical residue key with insertion code handling
                res_key = make_residue_key(chain.get_id(), residue)
                
                # Get pLDDT from CA atom B-factor (standard AlphaFold convention)
                try:
                    ca_atom = residue["CA"]
                    plddt = ca_atom.get_bfactor()
                except KeyError:
                    # Fall back to first atom if no CA (e.g., glycine)
                    atoms = list(residue.get_atoms())
                    if atoms:
                        plddt = atoms[0].get_bfactor()
                    else:
                        continue
                
                # Validate pLDDT is in expected range
                if not (0 <= plddt <= 100):
                    print(
                        f"Warning: Unexpected pLDDT value {plddt:.1f} at "
                        f"{res_key}",
                        file=sys.stderr
                    )
                
                plddt_scores[res_key] = plddt
    
    if not plddt_scores:
        raise ValueError("No pLDDT scores found in structure")
    
    return plddt_scores

def validate_chain_assignment(nanobody_chain, antigen_chain) -> Optional[str]:
    """
    Validate that chain assignment makes biological sense.
    
    Checks the size ratio of nanobody to antigen to detect potential
    chain swaps. A nanobody should be significantly smaller than the
    antigen (typical ratio < 0.5).
    
    Args:
        nanobody_chain: Putative nanobody chain
        antigen_chain: Putative antigen chain
        
    Returns:
        Warning message string, or None if validation passes
    """
    nb_size = count_protein_residues(nanobody_chain)
    ag_size = count_protein_residues(antigen_chain)
    
    # Check for potential chain swap: nanobody should be smaller
    size_ratio = nb_size / max(ag_size, 1)
    
    if size_ratio > 0.8:
        return (
            f"Chains have similar size (nanobody: {nb_size}, antigen: {ag_size}). "
            f"Chain assignment may be incorrect. Use --nanobody-chain and "
            f"--antigen-chain to specify explicitly."
        )
    
    if nb_size < NANOBODY_MIN_RESIDUES:
        return (
            f"Nanobody chain has only {nb_size} residues "
            f"(expected {NANOBODY_MIN_RESIDUES}-{NANOBODY_MAX_RESIDUES}). "
            f"This may be a fragment rather than a complete nanobody."
        )
    
    if nb_size > NANOBODY_MAX_RESIDUES:
        return (
            f"Nanobody chain has {nb_size} residues, "
            f"which is large for a nanobody (max {NANOBODY_MAX_RESIDUES}). "
            f"Check chain assignment."
        )
    
    return None


def identify_chains(

            )

    # Auto-detect: assume nanobody is the smaller chain

def get_interface_residues(
    structure: Structure,
    nanobody_chain_id: str,
    antigen_chain_id: str,
    distance_cutoff: float
) -> Tuple[List[str], List[str], int]:
    """
    Identify interface residues using NeighborSearch for efficiency.
    
    Uses Biopython's NeighborSearch with KD-tree for efficient spatial queries.
    Interface residues are defined as those with any heavy atom within the
    distance cutoff of any heavy atom in the other chain.
    
    Args:
        structure: Biopython Structure object
        nanobody_chain_id: Chain ID for the nanobody
        antigen_chain_id: Chain ID for the antigen
        distance_cutoff: Distance cutoff in Angstroms
        
    Returns:
        Tuple of:
        - List of residue keys for nanobody interface residues
        - List of residue keys for antigen interface residues
        - Number of inter-chain atom contacts within cutoff
    """
    model = next(structure.get_models())
    nb_chain = model[nanobody_chain_id]
    ag_chain = model[antigen_chain_id]
    
    # Build atom lists, storing residue key as an attribute on each atom
    # This avoids the need for separate mapping dictionaries
    nb_atoms = []
    ag_atoms = []
    
    for residue in nb_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(nanobody_chain_id, residue)
        for atom in residue:
            if atom.element != "H":  # Heavy atoms only
                atom.xtra['residue_key'] = res_key
                nb_atoms.append(atom)
    
    for residue in ag_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(antigen_chain_id, residue)
        for atom in residue:
            if atom.element != "H":
                atom.xtra['residue_key'] = res_key
                ag_atoms.append(atom)
    
    if not nb_atoms or not ag_atoms:
        return [], [], 0
    
    # Build NeighborSearch for antigen atoms
    neighbor_search = NeighborSearch(ag_atoms)
    
    # Find all nanobody atoms within distance cutoff of any antigen atom
    nb_interface_residues = set()
    ag_interface_residues = set()
    contact_count = 0
    
    for nb_atom in nb_atoms:
        close_atoms = neighbor_search.search(nb_atom.get_coord(), distance_cutoff)
        if close_atoms:
            nb_interface_residues.add(nb_atom.xtra['residue_key'])
            contact_count += len(close_atoms)
            for ag_atom in close_atoms:
                ag_interface_residues.add(ag_atom.xtra['residue_key'])
    
    return (
        list(nb_interface_residues),
        list(ag_interface_residues),
        contact_count
    )


def calculate_interface_plddt(

        overall_plddt = float(np.mean(list(plddt_scores.values())))
        
        # Identify interface residues
        nb_interface, ag_interface, num_contacts = get_interface_residues(
            structure, nb_chain, ag_chain, distance_cutoff

def generate_cross_variant_summary(
    results: List[Dict[str, object]],
    variant_map: Dict[str, Dict[str, str]]
) -> List[Dict[str, object]]:
    """
    Generate a ranked summary of nanobody designs across variants.
    
    For each nanobody design, calculates:
    - Mean interface pLDDT across all tested variants
    - Minimum interface pLDDT (worst-case performance)
    - Standard deviation (consistency across variants)
    
    Designs are ranked by a composite score: mean - std (penalizes inconsistency)
    
    Args:
        results: List of per-file result dictionaries
        variant_map: Dictionary mapping filenames to variant metadata
        
    Returns:
        List of summary dictionaries sorted by composite score (descending)
    """
    # Group results by nanobody design
    design_scores = defaultdict(list)
    
    for result in results:
        filename = result['filename']
        if filename in variant_map:
            design = variant_map[filename]['nanobody_design']
            variant = variant_map[filename]['variant']
            design_scores[design].append({
                'variant': variant,
                'interface_pLDDT': result['interface_pLDDT'],
                'filename': filename
            })
    
    if not design_scores:
        return []
    
    # Calculate summary statistics per design
    summaries = []
    for design, scores in design_scores.items():
        plddt_values = [s['interface_pLDDT'] for s in scores]
        mean_plddt = np.mean(plddt_values)
        min_plddt = min(plddt_values)
        std_plddt = np.std(plddt_values)
        
        # Composite score: high mean with low variance is best
        composite_score = mean_plddt - std_plddt
        
        # List variants and their scores
        variant_details = "; ".join(
            f"{s['variant']}={s['interface_pLDDT']:.1f}"
            for s in sorted(scores, key=lambda x: x['variant'])
        )
        
        summaries.append({
            'nanobody_design': design,
            'mean_interface_pLDDT': round(mean_plddt, 2),
            'min_interface_pLDDT': round(min_plddt, 2),
            'std_interface_pLDDT': round(std_plddt, 2),
            'composite_score': round(composite_score, 2),
            'num_variants_tested': len(scores),
            'variant_details': variant_details
        })
    
    # Sort by composite score (descending)
    summaries.sort(key=lambda x: x['composite_score'], reverse=True)
    
    return summaries

def make_residue_key(chain_id: str, residue) -> str:
    """

def extract_plddt_from_pdb(structure: Structure) -> Dict[str, float]:
    """
    Extract per-residue pLDDT scores from the B-factor column.
    
    AlphaFold-Multimer stores the predicted LDDT score (0-100) in the 
    B-factor column of all ATOM records. We use the CA atom's B-factor
    as the residue-level pLDDT.
    
    Args:
        structure: Biopython Structure object
        
    Returns:
        Dictionary mapping residue identifier (chain:residue_number) to pLDDT score
        
    Raises:
        ValueError: If no valid pLDDT scores are found or scores are out of range
    """
    plddt_scores = {}
    
    for model in structure:
        for chain in model:
            for residue in chain:
                # Skip heteroatoms (water, ligands)
                if residue.get_id()[0] != " ":
                    continue
                
                # Create canonical residue key with insertion code handling
                res_key = make_residue_key(chain.get_id(), residue)
                
                # Get pLDDT from CA atom B-factor (standard AlphaFold convention)
                try:
                    ca_atom = residue["CA"]
                    plddt = ca_atom.get_bfactor()
                except KeyError:
                    # Fall back to first atom if no CA (e.g., glycine)
                    atoms = list(residue.get_atoms())
                    if atoms:
                        plddt = atoms[0].get_bfactor()
                    else:
                        continue
                
                # Validate pLDDT is in expected range
                if not (0 <= plddt <= 100):
                    print(
                        f"Warning: Unexpected pLDDT value {plddt:.1f} at "
                        f"{res_key}",
                        file=sys.stderr
                    )
                
                plddt_scores[res_key] = plddt
    
    if not plddt_scores:
        raise ValueError("No pLDDT scores found in structure")
    
    return plddt_scores

def validate_chain_assignment(nanobody_chain, antigen_chain) -> Optional[str]:

    chain swaps. A nanobody should be significantly smaller than the

    size_ratio = nb_size / max(ag_size, 1)
    
    if size_ratio > 0.8:
        return (
            f"Chains have similar size (nanobody: {nb_size}, antigen: {ag_size}). "
            f"Chain assignment may be incorrect. Use --nanobody-chain and "
            f"--antigen-chain to specify explicitly."
        )
    
    if nb_size < NANOBODY_MIN_RESIDUES:
        return (
            f"Nanobody chain has only {nb_size} residues "
            f"(expected {NANOBODY_MIN_RESIDUES}-{NANOBODY_MAX_RESIDUES}). "
            f"This may be a fragment rather than a complete nanobody."
        )
    
    if nb_size > NANOBODY_MAX_RESIDUES:
        return (
            f"Nanobody chain has {nb_size} residues, "
            f"which is large for a nanobody (max {NANOBODY_MAX_RESIDUES}). "
            f"Check chain assignment."
        )
    
    return None


def identify_chains(
    structure: Structure,
    specified_nanobody: Optional[str] = None,
    specified_antigen: Optional[str] = None
) -> Tuple[str, str, Optional[str]]:
    """
    Identify nanobody and antigen chains with user override option.
    
    If chain IDs are specified, they are validated and returned.
    Otherwise, auto-detect by chain size (nanobody is smaller protein chain).
    
    Args:
        structure: Biopython Structure object
        specified_nanobody: Optional user-specified nanobody chain ID
        specified_antigen: Optional user-specified antigen chain ID
        
    Returns:
        Tuple of (nanobody_chain_id, antigen_chain_id, warning_message)
        
    Raises:
        ValueError: If chains cannot be identified or specified chains don't exist
    """
    model = next(structure.get_models())
    chains = list(model.get_chains())
    
    # Filter to only protein chains (ignore water, ligands)
    protein_chains = []
    for chain in chains:
        if count_protein_residues(chain) > 0:
            protein_chains.append(chain)
    
    if len(protein_chains) < 2:
        raise ValueError(
            f"Need at least 2 protein chains, found {len(protein_chains)} "
            f"in {len(chains)} total chains"
        )
    
    # Use user-specified chains if provided
    if specified_nanobody and specified_antigen:
        chain_ids = [c.get_id() for c in protein_chains]
        if specified_nanobody not in chain_ids:
            raise ValueError(
                f"Specified nanobody chain '{specified_nanobody}' not found. "
                f"Available chains: {chain_ids}"
            )
        if specified_antigen not in chain_ids:
            raise ValueError(
                f"Specified antigen chain '{specified_antigen}' not found. "
                f"Available chains: {chain_ids}"
            )
        
        nb_chain = model[specified_nanobody]
        ag_chain = model[specified_antigen]
        warning = validate_chain_assignment(nb_chain, ag_chain)
        return specified_nanobody, specified_antigen, warning
    
    # Auto-detect: assume nanobody is the smaller chain
    protein_chains.sort(key=lambda c: count_protein_residues(c))
    
    nb_chain = protein_chains[0]
    ag_chain = protein_chains[1]
    
    warning = validate_chain_assignment(nb_chain, ag_chain)
    
    # Warn about extra chains
    if len(protein_chains) > 2:
        extra_ids = [c.get_id() for c in protein_chains[2:]]
        extra_warning = (
            f"Found {len(protein_chains)} protein chains. "
            f"Using '{nb_chain.get_id()}' as nanobody and "
            f"'{ag_chain.get_id()}' as antigen. "
            f"Ignored chains: {extra_ids}. "
            f"Use --nanobody-chain and --antigen-chain to specify explicitly."
        )
        warning = f"{extra_warning} | {warning}" if warning else extra_warning
    
    return nb_chain.get_id(), ag_chain.get_id(), warning


def get_interface_residues(
    structure: Structure,
    nanobody_chain_id: str,
    antigen_chain_id: str,
    distance_cutoff: float
) -> Tuple[List[str], List[str], int]:
    """
    Identify interface residues using NeighborSearch for efficiency.
    
    Uses Biopython's NeighborSearch with KD-tree for efficient spatial queries.
    Interface residues are defined as those with any heavy atom within the
    distance cutoff of any heavy atom in the other chain.
    
    Args:
        structure: Biopython Structure object
        nanobody_chain_id: Chain ID for the nanobody
        antigen_chain_id: Chain ID for the antigen
        distance_cutoff: Distance cutoff in Angstroms
        
    Returns:
        Tuple of:
        - List of residue keys for nanobody interface residues
        - List of residue keys for antigen interface residues
        - Number of inter-chain atom contacts within cutoff
    """
    model = next(structure.get_models())
    nb_chain = model[nanobody_chain_id]
    ag_chain = model[antigen_chain_id]
    
    # Build atom lists, storing residue key as an attribute on each atom
    nb_atoms = []
    ag_atoms = []
    
    for residue in nb_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(nanobody_chain_id, residue)
        for atom in residue:
            if atom.element != "H":  # Heavy atoms only
                atom.xtra['residue_key'] = res_key
                nb_atoms.append(atom)
    
    for residue in ag_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(antigen_chain_id, residue)
        for atom in residue:
            if atom.element != "H":
                atom.xtra['residue_key'] = res_key
                ag_atoms.append(atom)
    
    if not nb_atoms or not ag_atoms:
        return [], [], 0
    
    # Build NeighborSearch for antigen atoms
    neighbor_search = NeighborSearch(ag_atoms)
    
    # Find all nanobody atoms within distance cutoff of any antigen atom
    nb_interface_residues = set()
    ag_interface_residues = set()
    contact_count = 0
    
    for nb_atom in nb_atoms:
        close_atoms = neighbor_search.search(nb_atom.get_coord(), distance_cutoff)
        if close_atoms:
            nb_interface_residues.add(nb_atom.xtra['residue_key'])
            contact_count += len(close_atoms)
            for ag_atom in close_atoms:
                ag_interface_residues.add(ag_atom.xtra['residue_key'])
    
    return (
        list(nb_interface_residues),
        list(ag_interface_residues),
        contact_count
    )


def calculate_interface_plddt(
    pdb_file: Path,
    distance_cutoff: float,
    specified_nanobody: Optional[str] = None,
    specified_antigen: Optional[str] = None
) -> Optional[Dict[str, Any]]:
    """
    Calculate interface pLDDT for a single PDB file.
    
    Returns None if processing fails (error printed to stderr).
    Otherwise returns a dictionary with results.
    
    Args:
        pdb_file: Path to PDB file
        distance_cutoff: Interface distance cutoff in Å
        specified_nanobody: Optional user-specified nanobody chain
        specified_antigen: Optional user-specified antigen chain
        
    Returns:
        Dictionary with keys:
        - 'filename': Name of PDB file
        - 'interface_pLDDT': Average pLDDT of interface residues (0-100)
        - 'overall_pLDDT': Average pLDDT of all residues (0-100)
        - 'num_interface_residues': Total residues at interface
        - 'num_nanobody_interface': Number of nanobody interface residues
        - 'num_antigen_interface': Number of antigen interface residues
        - 'nanobody_chain': Chain ID used for nanobody
        - 'antigen_chain': Chain ID used for antigen
        - 'num_contacts': Number of inter-chain atom contacts
        - 'warning': Any warnings about chain identification
        
    Returns None on failure.
    """
    try:
        # Parse PDB file
        parser = PDB.PDBParser(QUIET=True)
        structure = parser.get_structure("complex", str(pdb_file))
        
        # Extract pLDDT scores (raises ValueError if invalid)
        plddt_scores = extract_plddt_from_pdb(structure)
        
        # Identify chains (raises ValueError if can't identify)
        nb_chain, ag_chain, warning = identify_chains(
            structure, specified_nanobody, specified_antigen
        )
        
        # Calculate overall pLDDT
        overall_plddt = float(np.mean(list(plddt_scores.values())))
        
        # Identify interface residues
        nb_interface, ag_interface, num_contacts = get_interface_residues(
            structure, nb_chain, ag_chain, distance_cutoff
        )
        
        all_interface_residues = nb_interface + ag_interface
        
        if not all_interface_residues:
            print(
                f"Warning: {pdb_file.name}: No interface residues found "
                f"with distance cutoff {distance_cutoff}Å",
                file=sys.stderr
            )
            return None
        
        # Calculate interface pLDDT
        interface_plddt_values = [
            plddt_scores[res_key] for res_key in all_interface_residues
            if res_key in plddt_scores
        ]
        
        if not interface_plddt_values:
            print(
                f"Warning: {pdb_file.name}: No pLDDT scores found for interface residues",
                file=sys.stderr
            )
            return None
        
        interface_plddt = float(np.mean(interface_plddt_values))
        
        return {
            'filename': pdb_file.name,
            'interface_pLDDT': round(interface_plddt, 2),
            'overall_pLDDT': round(overall_plddt, 2),
            'num_interface_residues': len(all_interface_residues),
            'num_nanobody_interface': len(nb_interface),
            'num_antigen_interface': len(ag_interface),
            'nanobody_chain': nb_chain,
            'antigen_chain': ag_chain,
            'num_contacts': num_contacts,
            'warning': warning if warning else ''
        }
        
    except ValueError as e:
        print(f"Error: {pdb_file.name}: {e}", file=sys.stderr)
        return None
    except Exception as e:
        print(f"Error: {pdb_file.name}: Unexpected error: {e}", file=sys.stderr)
        return None


def generate_cross_variant_summary(
    results: List[Dict[str, Any]],
    variant_map: Dict[str, Dict[str, str]]
) -> List[Dict[str, Any]]:
    """
    Generate a ranked summary of nanobody designs across variants.
    
    For each nanobody design, calculates:
    - Mean interface pLDDT across all tested variants
    - Minimum interface pLDDT (worst-case performance)
    - Standard deviation (consistency across variants)
    
    Designs are ranked by a composite score: mean - std (penalizes inconsistency)
    
    Args:
        results: List of per-file result dictionaries
        variant_map: Dictionary mapping filenames to variant metadata
        
    Returns:
        List of summary dictionaries sorted by composite score (descending)
    """
    # Group results by nanobody design
    design_scores = defaultdict(list)
    
    for result in results:
        filename = result['filename']
        if filename in variant_map:
            design = variant_map[filename]['nanobody_design']
            variant = variant_map[filename]['variant']
            design_scores[design].append({
                'variant': variant,
                'interface_pLDDT': result['interface_pLDDT'],
                'filename': filename
            })
    
    if not design_scores:
        return []
    
    # Calculate summary statistics per design
    summaries = []
    for design, scores in design_scores.items():
        plddt_values = [s['interface_pLDDT'] for s in scores]
        mean_plddt = np.mean(plddt_values)
        min_plddt = min(plddt_values)
        std_plddt = np.std(plddt_values)
        
        # Composite score: high mean with low variance is best
        composite_score = mean_plddt - std_plddt
        
        # List variants and their scores
        variant_details = "; ".join(
            f"{s['variant']}={s['interface_pLDDT']:.1f}"
            for s in sorted(scores, key=lambda x: x['variant'])
        )
        
        summaries.append({
            'nanobody_design': design,
            'mean_interface_pLDDT': round(mean_plddt, 2),
            'min_interface_pLDDT': round(min_plddt, 2),
            'std_interface_pLDDT': round(std_plddt, 2),
            'composite_score': round(composite_score, 2),
            'num_variants_tested': len(scores),
            'variant_details': variant_details
        })
    
    # Sort by composite score (descending)
    summaries.sort(key=lambda x: x['composite_score'], reverse=True)
    
    return summaries
