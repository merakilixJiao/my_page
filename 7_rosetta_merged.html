"""
Run Rosetta binding energy calculation for nanobody-antigen complexes.

Executes Rosetta's rosetta_scripts with a provided XML protocol to calculate

energy data.

Requires Python 3.7+ for subprocess.run with text=True.

Usage:
    python run_rosetta_binding.py \\
        --pdb complex.pdb \\
        --xml bind_energy_calc.xml \\
        --rosetta /path/to/rosetta_scripts \\
        --output binding_results.sc

For robust results, run multiple decoys and average:
    for i in $(seq 1 10); do

import argparse

def validate_file(filepath: str, description: str) -> str:
    """
    Validate that a file exists and is readable.
    
    Args:
        filepath: Path to the file to validate
        description: Human-readable description of the file
        
    Returns:
        The validated filepath
        
    Raises:
        argparse.ArgumentTypeError: If file is invalid
    """
    if not os.path.isfile(filepath):
        raise argparse.ArgumentTypeError(
            f"{description} '{filepath}' does not exist or is not a file"
        )
    if not os.access(filepath, os.R_OK):
        raise argparse.ArgumentTypeError(
            f"{description} '{filepath}' is not readable"
        )
    return filepath


def validate_executable(filepath: str) -> str:
    """
    Validate that a Rosetta executable exists and is executable.
    
    Args:
        filepath: Path to the Rosetta executable
        
    Returns:
        The validated filepath
    """
    filepath = validate_file(filepath, "Rosetta executable")
    if not os.access(filepath, os.X_OK):
        raise argparse.ArgumentTypeError(
            f"Rosetta executable '{filepath}' is not executable"
        )
    return filepath


def validate_score_file_contains_ddg(filepath: str) -> bool:
    """
    Quick check if a score file contains a ddG column.
    
    Args:
        filepath: Path to the score file to validate
        
    Returns:
        True if file contains 'ddG' in its SCORE header
    """
    try:
        with open(filepath, 'r') as f:
            for line in f:
                if line.startswith('SCORE:') and 'ddG' in line:
                    return True
        return False
    except (IOError, OSError):
        return False

#!/usr/bin/env python3
"""
Aggregate Rosetta binding energy score files into a single sorted CSV.

Parses Rosetta score files (.sc) from a directory, extracts the binding 
energy (ddG) values computed by InterfaceAnalyzerMover, and outputs them 
sorted from most favorable (most negative ddG = strongest binding) to 
least favorable (most positive ddG = weakest binding).

For robust nanobody selection, run multiple decoys per variant and average
the ddG values before ranking. This script handles multiple decoys per file
by reporting the average ddG (more statistically robust than best single value).

Usage:
    python aggregate_binding_scores.py /path/to/score_directory -o results.csv
"""

import argparse
import csv
import glob
import math
import os
import sys
from typing import List, Tuple, Dict


def parse_score_file(filepath: str) -> Tuple[str, float]:
    """
    Parse a Rosetta score file to extract the binding energy (ddG).
    
    Uses the standard Rosetta score file format where the header line
    starts with 'SCORE:' followed by column names, and data lines contain
    the corresponding values. The 'ddG' column contains the binding energy.
    
    If multiple decoys are present in one file, averages all valid ddG values
    for a statistically robust estimate (more reliable than best single value).
    
    Args:
        filepath: Path to the Rosetta score file (.sc)
        
    Returns:
        Tuple of (filename, average_binding_energy_ddG)
        
    Raises:
        ValueError: If no valid ddG value can be found in the file,
                   or if all values are NaN/infinity
        IOError: If the file cannot be read
    """
    filename = os.path.basename(filepath)
    
    try:
        with open(filepath, 'r') as f:
            lines = f.readlines()
    except (IOError, OSError) as e:
        raise IOError(f"Cannot read file {filepath}: {e}")
    
    # Find header line and collect data lines
    header_line = None
    data_lines: List[str] = []
    
    for line in lines:
        line = line.strip()
        if not line or line.startswith('#'):
            continue
        if line.startswith('SCORE:'):
            header_line = line
        elif header_line is not None:
            data_lines.append(line)
    
    if header_line is None:
        raise ValueError(f"No SCORE header found in {filename}")
    
    # Parse header to find column indices
    header_parts = header_line.split()
    if header_parts[0] == 'SCORE:':
        header_parts = header_parts[1:]
    
    # Map column names to indices
    column_indices = {name: idx for idx, name in enumerate(header_parts)}
    
    if 'ddG' not in column_indices:
        raise ValueError(f"No ddG column found in {filename}")
    
    ddg_idx = column_indices['ddG']
    
    # Collect all valid ddG values (handle multiple decoys)
    valid_ddg_values: List[float] = []
    
    for data_line in data_lines:
        parts = data_line.split()
        if len(parts) <= ddg_idx:
            continue
        
        try:
            ddg_value = float(parts[ddg_idx])
        except (ValueError, IndexError):
            continue
        
        # Filter out NaN and infinity values
        if math.isfinite(ddg_value):
            valid_ddg_values.append(ddg_value)
    
    if not valid_ddg_values:
        raise ValueError(
            f"No valid finite ddG values found in {filename}. "
            f"All values were NaN, infinite, or missing. "
            f"This may indicate numerical instability in the calculation."
        )
    
    # Average all valid decoy values for robust estimate
    average_ddg = sum(valid_ddg_values) / len(valid_ddg_values)
    
    # Warn if multiple decoys had high variance
    if len(valid_ddg_values) > 1:
        variance = sum((x - average_ddg) ** 2 for x in valid_ddg_values) / len(valid_ddg_values)
        std_dev = math.sqrt(variance)
        if std_dev > 2.0:  # More than 2 kcal/mol standard deviation
            print(f"Warning: {filename} has high variance in ddG values "
                  f"(std_dev = {std_dev:.2f} kcal/mol). Consider running "
                  f"more decoys for reliable estimates.", file=sys.stderr)
    
    return filename, average_ddg


def aggregate_score_files(input_dir: str) -> List[Tuple[str, float]]:
    """
    Aggregate all Rosetta score files in a directory.
    
    Args:
        input_dir: Directory containing .sc files
        
    Returns:
        List of (filename, binding_energy) tuples, sorted by binding energy
        from most favorable (most negative ddG = strongest binding) to
        least favorable (most positive ddG = weakest binding)
        
    Raises:
        FileNotFoundError: If no .sc files are found
        ValueError: If no valid binding energies could be extracted
    """
    # Find all .sc files in the directory
    score_files = glob.glob(os.path.join(input_dir, '*.sc'))
    
    if not score_files:
        raise FileNotFoundError(
            f"No .sc files found in directory '{input_dir}'. "
            f"Ensure Rosetta has been run and output .sc files exist."
        )
    
    results: Dict[str, float] = {}  # Track by filename to detect duplicates
    failed_files: List[str] = []
    
    for filepath in sorted(score_files):  # Sort for reproducibility
        try:
            result = parse_score_file(filepath)
            filename, ddg = result
            
            # Check for duplicate filenames
            if filename in results:
                print(f"Warning: Duplicate filename '{filename}' detected. "
                      f"Keeping first occurrence from sorted order.",
                      file=sys.stderr)
                continue
            
            results[filename] = ddg
            
        except (ValueError, IOError) as e:
            failed_files.append(os.path.basename(filepath))
            print(f"Warning: Failed to parse {os.path.basename(filepath)}: {e}", 
                  file=sys.stderr)
    
    # Report parsing failures
    if failed_files:
        print(f"\nWarning: {len(failed_files)} files could not be parsed:", 
              file=sys.stderr)
        for fname in failed_files:
            print(f"  {fname}", file=sys.stderr)
    
    if not results:
        raise ValueError(
            "No valid binding energy values could be extracted from any "
            "score files. Check that Rosetta was run with InterfaceAnalyzerMover "
            "and that ddG values are present in the output."
        )
    
    # Convert to list and sort by ddG ascending (most negative = strongest binding first)
    sorted_results = sorted(results.items(), key=lambda x: x[1])
    
    return sorted_results


def write_csv(results: List[Tuple[str, float]], output_path: str) -> None:
    """
    Write results to a CSV file sorted from best to worst binder.
    
    Binding energy interpretation (typical ranges for nanobody-spike interactions):
    - ddG < -10 kcal/mol: Strong binding candidate for experimental testing
    - ddG between -10 and -5: Moderate binding, consider for validation
    - ddG > 0: Unfavorable binding, likely non-binder
    
    These thresholds are based on typical nanobody binding energies reported
    in the literature (e.g., Schoof et al. 2020, Cell; Xiang et al. 2020,
    Science). Always validate experimentally and include negative controls.
    
    Args:
        results: List of (filename, binding_energy) tuples, already sorted
                 from most favorable (most negative) to least favorable
        output_path: Path to the output CSV file
    """
    with open(output_path, 'w', newline='') as csvfile:
        writer = csv.writer(csvfile)
        writer.writerow(['rank', 'filename', 'binding_energy_ddG_kcal_per_mol'])
        
        for rank, (filename, ddg) in enumerate(results, 1):
            writer.writerow([rank, filename, f"{ddg:.3f}"])
    
    print(f"Results written to {output_path}")
    print(f"Total entries: {len(results)}")
    
    # Provide summary statistics
    ddg_values = [r[1] for r in results]
    print(f"Binding energy range: {min(ddg_values):.2f} to {max(ddg_values):.2f} kcal/mol")
    print(f"Median binding energy: {sorted(ddg_values)[len(ddg_values)//2]:.2f} kcal/mol")
    
    # Highlight top candidates
    print(f"\nTop 5 nanobody candidates (strongest binding, most negative ddG):")
    for rank, (filename, ddg) in enumerate(results[:5], 1):
        print(f"  Rank {rank}: {filename} (ddG = {ddg:.3f} kcal/mol)")
    
    print(f"\nInterpretation guidance for nanobody selection:")
    print(f"  ddG < -10 kcal/mol: Strong candidate for experimental testing")
    print(f"  ddG -10 to -5 kcal/mol: Moderate candidate, consider validation")
    print(f"  ddG > 0 kcal/mol: Unfavorable, likely non-binder")
    print(f"  Always run a negative control (non-binding nanobody) to establish baseline")


def main() -> None:
    """Main entry point for the script."""
    parser = argparse.ArgumentParser(
        description='Aggregate Rosetta binding energy score files into sorted CSV',
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
Examples:
    # Basic usage
    python aggregate_binding_scores.py ./rosetta_scores/ -o binding_results.csv
    
    # For robust selection, average multiple decoys:
    # 1. Run Rosetta 10 times per variant using run_rosetta_binding.py
    # 2. Run this script to aggregate all results
    # 3. Top-ranked candidates have most favorable (most negative) ddG values
        
Sorting: Results are sorted from strongest binding (most negative ddG) to
weakest binding (most positive ddG). This corresponds to highest affinity first.
        """
    )
    
    parser.add_argument(
        'input_dir',
        type=str,
        help='Directory containing Rosetta score files (.sc)'
    )
    
    parser.add_argument(
        '-o', '--output',
        type=str,
        default='binding_energy_results.csv',
        help='Output CSV file path (default: binding_energy_results.csv)'
    )
    
    args = parser.parse_args()
    
    # Validate input directory
    if not os.path.isdir(args.input_dir):
        print(f"Error: '{args.input_dir}' is not a valid directory", 
              file=sys.stderr)
        sys.exit(1)
    
    # Aggregate scores
    try:
        results = aggregate_score_files(args.input_dir)
    except (FileNotFoundError, ValueError) as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
    
    # Write results to CSV
    write_csv(results, args.output)


if __name__ == '__main__':
    main()

    1. Minimizes the complex

    3. Computes ΔG = E_complex - (E_nanobody + E_antigen)

    score file. More negative values indicate stronger predicted binding.

        rosetta_scripts.default.linuxgccrelease \
            -parser:protocol this_file.xml \
            -parser:script_vars nanobody_chain=A antigen_chain=B \

    - Remove water molecules and ligands
    - Ensure standard residue naming
    - Use -in:ignore_waters command-line flag for water removal

        <RestrictToRepacking name="repack_only"/>

        This is a standard minimization protocol that maintains overall structure.
        Missing sidechains are handled by the -packing:pack_missing_sidechains

#!/usr/bin/env python3
"""
Process Rosetta binding energy score files and output sorted CSV.

This script parses Rosetta score files (.sc format) from InterfaceAnalyzer
calculations and extracts the dG_separated (binding energy) values. Results
are sorted from strongest binding (most negative) to weakest binding (most
positive) and written to a CSV file.

For each input score file, the script reports the BEST (lowest/most negative)
dG_separated value. This ensures one row per file as requested.

Usage:
    python process_rosetta_scores.py /path/to/score_directory -o sorted_scores.csv
    python process_rosetta_scores.py . -o results.csv --weakest-first
    python process_rosetta_scores.py /path/to/scores -o best.csv --best-per-file

Score interpretation (dG_separated in Rosetta Energy Units, REU):
    More negative = stronger binding
    dG < -10.0 : Very strong binding
    dG < -5.0  : Strong binding
    dG < -2.0  : Moderate binding
    dG < 0.0   : Weak binding
    dG > 0.0   : No favorable binding predicted
"""

import argparse
import csv
import glob
import os
import sys
from typing import Dict, List, Tuple


def parse_rosetta_score_file(filepath: str) -> List[Tuple[str, float]]:
    """
    Parse a Rosetta score file and extract dG_separated binding energies.

    Rosetta score files use columnar format with a header line starting with
    'SCORE:' followed by column names, then data lines with values. This
    function identifies the 'dG_separated' column and extracts all values.

    Args:
        filepath: Path to the Rosetta score file (.sc format)

    Returns:
        List of (description, dG_separated) tuples, one per decoy.
        Description includes source filename and decoy identifier.

    Raises:
        ValueError: If file cannot be parsed or no dG_separated column found
        IOError: If file cannot be read
    """
    filename = os.path.basename(filepath)
    results: List[Tuple[str, float]] = []

    try:
        with open(filepath, 'r') as f:
            lines = f.readlines()
    except IOError as e:
        raise IOError(f"Cannot read file '{filepath}': {e}")

    if not lines:
        raise ValueError(f"Empty score file: {filepath}")

    # Find header line (starts with "SCORE:" followed by column names)
    header_line = None
    header_index = -1
    for i, line in enumerate(lines):
        stripped = line.strip()
        if stripped.startswith("SCORE:") and not stripped.startswith("SCORE: "):
            header_line = stripped
            header_index = i
            break

    if header_line is None:
        raise ValueError(f"No valid Rosetta header found in '{filepath}'")

    # Parse column names from header
    # Format: "SCORE:  col1  col2  col3  ..."
    columns = header_line.split()[1:]  # Remove "SCORE:" prefix

    # Find column indices for dG_separated and description
    dg_col = -1
    desc_col = -1
    for i, col in enumerate(columns):
        if col == "dG_separated":
            dg_col = i
        elif col == "description":
            desc_col = i

    if dg_col == -1:
        raise ValueError(
            f"No 'dG_separated' column found in '{filepath}'. "
            f"Available columns: {columns}. "
            "Ensure InterfaceAnalyzer was used in the Rosetta protocol."
        )

    if desc_col == -1:
        # Last column is often description if not labeled
        desc_col = len(columns) - 1

    # Parse data lines (start with "SCORE: " followed by values)
    for line in lines[header_index + 1:]:
        stripped = line.strip()
        if not stripped.startswith("SCORE:"):
            continue

        # Remove "SCORE:" prefix and split by whitespace
        parts = stripped.split()[1:]

        if len(parts) <= max(dg_col, desc_col):
            continue  # Skip lines with insufficient columns

        try:
            dg_value = float(parts[dg_col])
            decoy_desc = parts[desc_col]
            # Create unique identifier combining filename and decoy description
            full_desc = f"{filename}:{decoy_desc}"
            results.append((full_desc, dg_value))
        except (ValueError, IndexError):
            continue  # Skip malformed lines

    if not results:
        raise ValueError(
            f"No valid data lines found in '{filepath}'. "
            "The calculation may have failed - check rosetta_run.log."
        )

    return results


def get_best_score_per_file(
    results: List[Tuple[str, float]]
) -> Dict[str, float]:
    """
    Aggregate multiple decoys per file into a single best score per file.

    For files with multiple decoys, this selects the lowest (most negative)
    dG_separated value, representing the strongest predicted binding.

    Args:
        results: List of (description, dG_separated) tuples from parsing

    Returns:
        Dictionary mapping base filename to best (lowest) dG_separated value
    """
    file_scores: Dict[str, List[float]] = {}
    
    for desc, dg in results:
        # Extract base filename from description (format: "filename:decoy_desc")
        base_filename = desc.split(":")[0]
        if base_filename not in file_scores:
            file_scores[base_filename] = []
        file_scores[base_filename].append(dg)
    
    # For each file, keep the best (lowest/most negative) score
    best_scores: Dict[str, float] = {}
    for filename, scores in file_scores.items():
        best_scores[filename] = min(scores)  # Most negative = strongest binding
    
    return best_scores


def process_score_directory(
    input_dir: str,
    best_per_file: bool = True
) -> List[Tuple[str, float]]:
    """
    Process all Rosetta score files in a directory.

    Args:
        input_dir: Directory containing Rosetta score files (*.sc extension)
        best_per_file: If True, report best score per file (one row per file).
                      If False, report all decoys (one row per decoy).

    Returns:
        List of (identifier, dG_separated) tuples

    Raises:
        FileNotFoundError: If input directory doesn't exist
        ValueError: If no valid score files found
    """
    if not os.path.isdir(input_dir):
        raise FileNotFoundError(f"Directory not found: '{input_dir}'")

    # Find all score files with common Rosetta extensions
    score_files = []
    for ext in ["*.sc", "*.fasc", "*.score"]:
        score_files.extend(glob.glob(os.path.join(input_dir, ext)))

    # Remove duplicates while preserving order
    seen = set()
    unique_files = []
    for f in score_files:
        if f not in seen:
            seen.add(f)
            unique_files.append(f)

    if not unique_files:
        raise ValueError(
            f"No Rosetta score files (*.sc, *.fasc, *.score) found in '{input_dir}'"
        )

    # Parse each file
    all_results: List[Tuple[str, float]] = []
    errors: List[Tuple[str, str]] = []

    for filepath in sorted(unique_files):
        try:
            file_results = parse_rosetta_score_file(filepath)
            all_results.extend(file_results)
        except (ValueError, IOError) as e:
            errors.append((filepath, str(e)))

    # Report parsing errors but continue with valid data
    if errors:
        print("Warning: Could not parse the following files:", file=sys.stderr)
        for filepath, error in errors:
            print(f"  {filepath}: {error}", file=sys.stderr)
        print("Check rosetta_run.log files for calculation errors.", file=sys.stderr)

    if not all_results:
        raise ValueError(
            f"No valid binding energy data could be extracted from any files "
            f"in '{input_dir}'."
        )

    # Aggregate per file if requested (default behavior)
    if best_per_file:
        best_scores = get_best_score_per_file(all_results)
        # Convert back to sorted list of tuples
        final_results = [(fname, score) for fname, score in best_scores.items()]
        print(f"Aggregated {len(all_results)} decoys into {len(final_results)} "
              f"best-per-file entries.")
        return final_results
    else:
        return all_results


def write_sorted_csv(
    results: List[Tuple[str, float]],
    output_path: str,
    strongest_first: bool = True
) -> None:
    """
    Write results to a CSV file, sorted by binding energy.

    Args:
        results: List of (identifier, dG_separated) tuples
        output_path: Path to output CSV file
        strongest_first: If True (default), sort strongest binders first
                        (most negative dG). If False, sort weakest first.
    """
    # Sort by dG_separated value
    # Default: ascending order (most negative = strongest = first)
    sorted_results = sorted(results, key=lambda x: x[1], reverse=not strongest_first)

    with open(output_path, 'w', newline='') as csvfile:
        writer = csv.writer(csvfile)
        writer.writerow(['filename', 'dG_separated_REU'])

        for identifier, dg in sorted_results:
            writer.writerow([identifier, f"{dg:.3f}"])

    print(f"Wrote {len(sorted_results)} entries to '{output_path}'")

    # Print summary statistics
    dg_values = [dg for _, dg in results]
    if dg_values:
        n_strong = sum(1 for dg in dg_values if dg < -5.0)
        print(f"Binding energy range: {min(dg_values):.3f} to {max(dg_values):.3f} REU")
        print(f"Mean binding energy: {sum(dg_values) / len(dg_values):.3f} REU")
        print(f"Strong binders (dG < -5.0): {n_strong}/{len(dg_values)}")

#!/usr/bin/env python3
"""
Aggregate Rosetta binding energy score files into a sorted CSV.

This script reads multiple Rosetta score files (with .sc extension) from a specified
directory, extracts the binding energy (dG_separated from InterfaceAnalyzer), and
outputs a CSV file sorted by binding energy.

Binding energy interpretation (in Rosetta Energy Units, REU):
  - More negative dG_separated values indicate stronger binding (higher affinity)
  - More positive dG_separated values indicate weaker binding (lower affinity)
  - Typical effective nanobodies have dG_separated < -5 REU

Sorting behavior:
  Default (agenda requirement: "highest to lowest score"):
    "Highest score" refers to strongest binding affinity, which corresponds to
    the most negative dG_separated values. So default sort is from most negative
    (strongest binding) to most positive (weakest binding).
  
  With --reverse: Sort from most positive (weakest binding) to most negative
                  (strongest binding).

Usage:
    python aggregate_rosetta_scores.py <input_directory> [output_csv] [--reverse]

Examples:
    python aggregate_rosetta_scores.py ./rosetta_output/
    python aggregate_rosetta_scores.py ./rosetta_output/ binding_results.csv
    python aggregate_rosetta_scores.py ./rosetta_output/ --reverse  # Weakest first
"""

import argparse
import csv
import os
import sys
from typing import List, Tuple


# Rosetta score file format constants
SCORE_LINE_PREFIX = "SCORE:"
PRIMARY_BINDING_COLUMN = "dG_separated"
FALLBACK_COLUMN = "total_score"


def parse_rosetta_score_file(filepath: str) -> Tuple[str, float]:
    """
    Parse a Rosetta score file and extract the binding energy.

    The function looks for the 'dG_separated' column (from InterfaceAnalyzerMover)
    as the primary binding energy metric. If not found, it falls back to 'total_score'.
    If neither is found, it raises a ValueError with diagnostic information.

    If the score file contains multiple data lines (e.g., from -nstruct > 1),
    a warning is printed and only the first data line is used.

    Args:
        filepath: Path to the Rosetta score file (.sc extension)

    Returns:
        Tuple of (filename, binding_energy_score)

    Raises:
        ValueError: If the score file format is invalid or no binding energy found.
        IOError: If the file cannot be read.
    """
    filename = os.path.basename(filepath)

    # Read all lines from the file
    try:
        with open(filepath, 'r') as f:
            lines = f.readlines()
    except IOError as e:
        raise IOError(f"Cannot read file '{filepath}': {e}")

    # Filter for SCORE: lines (Rosetta's standard output format)
    score_lines = [line.strip() for line in lines if line.strip().startswith(SCORE_LINE_PREFIX)]

    if len(score_lines) < 2:
        raise ValueError(
            f"Invalid Rosetta score file '{filepath}': "
            f"Expected at least 2 SCORE: lines (header + data), found {len(score_lines)}. "
            f"Ensure this is a valid Rosetta .sc output file."
        )

    # Warn if multiple data lines exist (multiple structures in single file)
    num_data_lines = len(score_lines) - 1  # Subtract header
    if num_data_lines > 1:
        print(f"  [WARN] {filename}: Contains {num_data_lines} data lines. "
              f"Using only the first data line.", file=sys.stderr)

    # First SCORE: line is the header (column names)
    header_line = score_lines[0]
    header_terms = header_line[len(SCORE_LINE_PREFIX):].strip().split()

    # Determine which column to use for binding energy
    # Try dG_separated first (from InterfaceAnalyzer), then total_score
    binding_col_idx = None
    for col_name in [PRIMARY_BINDING_COLUMN, FALLBACK_COLUMN]:
        if col_name in header_terms:
            binding_col_idx = header_terms.index(col_name)
            binding_col_name = col_name
            break

    if binding_col_idx is None:
        # Provide helpful error message with available columns
        available_cols = ", ".join(header_terms[:15])  # Show first 15 columns
        raise ValueError(
            f"Cannot find binding energy column in '{filepath}'. "
            f"Looked for '{PRIMARY_BINDING_COLUMN}' and '{FALLBACK_COLUMN}'. "
            f"Available columns (first 15): {available_cols}. "
            f"Ensure the Rosetta protocol includes InterfaceAnalyzerMover."
        )

    # Second SCORE: line contains the data (first structure's scores)
    data_line = score_lines[1]
    data_terms = data_line[len(SCORE_LINE_PREFIX):].strip().split()

    if len(data_terms) <= binding_col_idx:
        raise ValueError(
            f"Data line in '{filepath}' has {len(data_terms)} columns, "
            f"but expected at least {binding_col_idx + 1} columns "
            f"for column '{binding_col_name}' at index {binding_col_idx}."
        )

    # Parse the score value
    try:
        score = float(data_terms[binding_col_idx])
    except ValueError as e:
        raise ValueError(
            f"Cannot parse '{binding_col_name}' value from '{filepath}': "
            f"got '{data_terms[binding_col_idx]}' at column {binding_col_idx}. "
            f"Error: {e}"
        )

    return filename, score


def aggregate_scores(input_dir: str, output_csv: str, reverse_sort: bool = False) -> None:
    """
    Aggregate all Rosetta score files from a directory into a sorted CSV.

    Default sort (reverse_sort=False): strongest binding first (most negative scores).
    This follows the scientific convention where "highest score" means highest
    binding affinity (most negative dG).

    Args:
        input_dir: Directory containing Rosetta .sc score files.
        output_csv: Path to the output CSV file.
        reverse_sort: If True, sort weakest binding first (most positive scores).
                      Default (False) sorts strongest binding first.
    """
    # Validate input directory
    if not os.path.isdir(input_dir):
        print(f"Error: '{input_dir}' is not a valid directory.", file=sys.stderr)
        sys.exit(1)

    # Find all .sc files in the input directory
    score_files: List[str] = [
        os.path.join(input_dir, f) for f in os.listdir(input_dir)
        if f.endswith('.sc') and os.path.isfile(os.path.join(input_dir, f))
    ]

    if not score_files:
        print(f"Error: No .sc files found in '{input_dir}'.", file=sys.stderr)
        sys.exit(1)

    print(f"Found {len(score_files)} score file(s) to process...")

    # Parse each score file
    parsed_scores: List[Tuple[str, float]] = []
    errors: List[str] = []

    for filepath in sorted(score_files):  # Sort for deterministic order
        try:
            filename, score = parse_rosetta_score_file(filepath)
            parsed_scores.append((filename, score))
            print(f"  [OK] {filename}: {score:.3f}")
        except (ValueError, IOError) as e:
            errors.append(str(e))
            print(f"  [ERR] {os.path.basename(filepath)}: {e}", file=sys.stderr)

    if not parsed_scores:
        print("Error: No valid scores could be parsed from any file.", file=sys.stderr)
        if errors:
            print("Errors encountered:", file=sys.stderr)
            for err in errors:
                print(f"  - {err}", file=sys.stderr)
        sys.exit(1)

    # Sort by score
    # Default (reverse_sort=False): Most negative first (strongest binding first)
    # This matches scientific convention for "highest score" = highest affinity
    # If reverse_sort=True: Most positive first (weakest binding first)
    parsed_scores.sort(key=lambda x: x[1], reverse=not reverse_sort)

    # Ensure output directory exists
    output_dir = os.path.dirname(output_csv)
    if output_dir and not os.path.exists(output_dir):
        os.makedirs(output_dir, exist_ok=True)

    # Write to CSV
    try:
        with open(output_csv, 'w', newline='') as csvfile:
            writer = csv.writer(csvfile)
            writer.writerow(['filename', 'binding_score'])
            writer.writerows(parsed_scores)
    except IOError as e:
        print(f"Error: Cannot write to '{output_csv}': {e}", file=sys.stderr)
        sys.exit(1)

    # Summary statistics
    scores_only = [score for _, score in parsed_scores]
    min_score = min(scores_only)
    max_score = max(scores_only)

    sort_description = "strongest binding first (most negative)" if not reverse_sort else "weakest binding first (most positive)"
    print(f"\nSuccessfully wrote {len(parsed_scores)} entries to '{output_csv}'.")
    print(f"Score range: {min_score:.3f} (strongest binding) to {max_score:.3f} (weakest binding)")
    print(f"Sorted: {sort_description}")

    if errors:
        print(f"\nWarning: {len(errors)} file(s) could not be parsed. See errors above.", file=sys.stderr)

"""

calculations and extracts the dG_separated binding energy column.

Output: CSV file sorted by binding energy (most negative = strongest binder first)

Binding energy interpretation:
  - dG_separated from InterfaceAnalyzer (kcal/mol)
  - More negative = stronger binding
  - Typical nanobody-spike protein: -10 to -25 kcal/mol
  - Values near zero or positive indicate no binding

For reproducibility with multiple runs (-nstruct N):
  - This script uses the first SCORE data line from each file
  - For multiple runs, average results across output files
  - Score files from different random seeds should give consistent values
  (within ~0.5-1.0 kcal/mol for well-behaved complexes)

Usage:
  python process_scores.py <input_directory> [output_csv]
  
Example:
  python process_scores.py binding_results/ sorted_binders.csv
"""

import argparse
import csv
import os
import sys
from typing import List, Tuple


def parse_rosetta_score_file(filepath: str) -> Tuple[str, float]:
    """
    Parse a Rosetta score file to extract binding energy (dG_separated).
    
    Rosetta score file format (from InterfaceAnalyzer):
      SEQUENCE: ...
      SCORE: total_score dG_separated dG_cross ...  (header line)
      SCORE:   -45.3    -12.5         -8.2          (data line)
    
    Note: If multiple SCORE data lines exist (from -nstruct > 1),
    this function uses the FIRST data line. For multi-run analysis,
    process each output file separately.
    
    Args:
        filepath: Path to .sc score file
        
    Returns:
        Tuple of (complex_name, binding_energy)
        
    Raises:
        ValueError: If dG_separated column not found or cannot be parsed
        FileNotFoundError: If file doesn't exist
    """
    if not os.path.exists(filepath):
        raise FileNotFoundError(f"Score file not found: {filepath}")
    
    # Complex name from filename (remove .sc extension)
    complex_name = os.path.splitext(os.path.basename(filepath))[0]
    
    with open(filepath, 'r') as f:
        lines = f.readlines()
    
    if not lines:
        raise ValueError(f"Empty score file: {filepath}")
    
    # Find header line and first data line (both start with "SCORE:")
    header_line = None
    first_data_line = None
    
    for line in lines:
        stripped = line.strip()
        if not stripped.startswith('SCORE:'):
            continue
        if header_line is None:
            header_line = stripped
        elif first_data_line is None:
            first_data_line = stripped
    
    if header_line is None:
        raise ValueError(f"No SCORE header found in: {filepath}")
    
    if first_data_line is None:
        raise ValueError(f"No SCORE data found in: {filepath}")
    
    # Parse header to find dG_separated column index
    header_fields = header_line.split()
    column_names = header_fields[1:]  # Remove "SCORE:" label
    
    try:
        dg_col_index = column_names.index('dG_separated')
    except ValueError:
        raise ValueError(
            f"Column 'dG_separated' not found in {filepath}. "
            f"Available columns: {', '.join(column_names[:10])}..."
        )
    
    # Parse first data line
    data_fields = first_data_line.split()
    data_values = data_fields[1:]  # Remove "SCORE:" label
    
    if dg_col_index >= len(data_values):
        raise ValueError(
            f"Data line too short in {filepath}. "
            f"Expected column {dg_col_index}, found {len(data_values)} values."
        )
    
    try:
        binding_energy = float(data_values[dg_col_index])
    except ValueError:
        raise ValueError(
            f"Could not parse binding energy from '{data_values[dg_col_index]}' "
            f"in {filepath}"
        )
    
    return complex_name, binding_energy


def process_directory(input_dir: str) -> List[Tuple[str, float]]:
    """
    Process all .sc files in a directory.
    
    Args:
        input_dir: Directory containing .sc score files
        
    Returns:
        List of (complex_name, binding_energy) tuples
        
    Raises:
        NotADirectoryError: If input_dir is not a valid directory
    """

        raise NotADirectoryError(f"Not a directory: {input_dir}")
    
    results = []
    
    # Find all .sc files (excluding hidden files)
    score_files = sorted([
        f for f in os.listdir(input_dir) 
        if f.endswith('.sc') and not f.startswith('.')
    ])
    
    if not score_files:
        print(f"Warning: No .sc files found in {input_dir}", file=sys.stderr)
        return results
    
    # Process each score file
    for filename in score_files:
        filepath = os.path.join(input_dir, filename)
        try:

#!/usr/bin/env python3
"""
Process Rosetta binding energy score files and output sorted results as CSV.

Reads all .sc files from a directory, extracts binding energy (ddG) values,
and outputs a CSV sorted from strongest to weakest binding
(most negative to most positive ddG).

Handles multiple data lines per file by taking the mean ddG value,
with a warning if multiple lines are found.

Usage:
    python process_rosetta_scores.py <input_directory> <output_csv>
"""

import argparse
import csv
import glob
import os
import re
import sys
from math import isfinite
from statistics import mean
from typing import List, Tuple


def parse_rosetta_score_file(filepath: str) -> Tuple[str, float]:
    """
    Parse a Rosetta score file to extract the binding energy (ddG).

    If multiple data lines are present (multiple decoys), returns the mean ddG.
    This handles cases where Rosetta generates multiple output models.

    Args:
        filepath: Path to Rosetta .sc score file

    Returns:
        Tuple of (basename, mean_ddG_value)

    Raises:
        ValueError: If parsing fails, ddG column not found, or all values are invalid
    """
    basename = os.path.basename(filepath)

    with open(filepath, 'r') as f:
        lines = f.readlines()

    if len(lines) < 2:
        raise ValueError(f"Insufficient lines ({len(lines)}) in {basename}. "
                        f"Expected at least a header and one data line.")

    # Find all SCORE lines (header and data lines)
    score_lines = []
    for line in lines:
        stripped = line.strip()
        if stripped.startswith('SCORE:'):
            score_lines.append(stripped)

    if len(score_lines) < 2:
        raise ValueError(f"Found {len(score_lines)} SCORE line(s) in {basename}. "
                        f"Need at least 2 (header + data).")

    # First SCORE line is the header, rest are data lines
    header_line = score_lines[0]
    data_lines = score_lines[1:]

    # Parse column headers
    headers = re.split(r'\s+', header_line[6:].strip())

    # Find ddG column
    try:
        ddg_idx = headers.index('ddG')
    except ValueError:
        raise ValueError(f"No 'ddG' column in {basename}. "
                        f"Available columns: {', '.join(headers)}")

    # Extract ddG values from all data lines
    ddg_values = []
    for i, data_line in enumerate(data_lines):
        values = re.split(r'\s+', data_line[6:].strip())
        if len(values) != len(headers):
            raise ValueError(f"Data line {i+1} in {basename} has {len(values)} "
                            f"columns, expected {len(headers)}.")

        try:
            ddg = float(values[ddg_idx])
        except (ValueError, IndexError):
            raise ValueError(f"Cannot parse ddG value in data line {i+1} "
                            f"of {basename}: '{values[ddg_idx]}'")

        if not isfinite(ddg):
            print(f"Warning: Non-finite ddG ({ddg}) found in data line "
                  f"{i+1} of {basename}, skipping this line.", file=sys.stderr)
            continue

        ddg_values.append(ddg)

    if not ddg_values:
        raise ValueError(f"No valid ddG values found in {basename}")

    # If multiple data lines, take the mean and warn
    if len(ddg_values) > 1:
        print(f"Note: {basename} contains {len(ddg_values)} data lines. "
              f"Using mean ddG = {mean(ddg_values):.4f}.", file=sys.stderr)

    # Use mean of all ddG values
    return basename, mean(ddg_values)


def main() -> int:
    """Main entry point. Returns 0 on success, 1 on failure."""
    parser = argparse.ArgumentParser(
        description=(
            "Process Rosetta .sc score files and output sorted CSV. "
            "Sorts from strongest to weakest binding "
            "(most negative to most positive ddG). "
            "Multiple data lines per file are averaged."
        )
    )
    parser.add_argument('input_dir', help="Directory containing .sc files")
    parser.add_argument('output_csv', help="Path to output CSV file")
    args = parser.parse_args()

    # Validate input directory
    if not os.path.isdir(args.input_dir):
        print(f"Error: Directory not found: {args.input_dir}", file=sys.stderr)
        return 1

    # Check if output file already exists
    if os.path.exists(args.output_csv):
        response = input(f"Warning: {args.output_csv} already exists. "
                        f"Overwrite? [y/N]: ").strip().lower()
        if response != 'y':
            print("Aborted.", file=sys.stderr)
            return 1

    # Find all .sc files
    score_files = sorted(glob.glob(os.path.join(args.input_dir, '*.sc')))
    if not score_files:
        print(f"Error: No .sc files found in {args.input_dir}", file=sys.stderr)
        return 1

    print(f"Found {len(score_files)} score files to process.")

    # Parse all score files
    results: List[Tuple[str, float]] = []
    errors = 0
    for filepath in score_files:
        try:
            results.append(parse_rosetta_score_file(filepath))
        except ValueError as e:
            print(f"Error: {e}", file=sys.stderr)
            errors += 1

    if not results:
        print("Error: No valid score files could be parsed.", file=sys.stderr)
        return 1

    # Sort from most negative to most positive (strongest to weakest binding)
    results.sort(key=lambda x: x[1])

    # Write CSV
    with open(args.output_csv, 'w', newline='') as f:
        writer = csv.writer(f)
        # Write metadata as a comment line (RFC 4180 allows this)
        writer.writerow([
            '# Sorted from strongest binding (most negative ddG) '
            'to weakest binding (most positive ddG)'
        ])
        writer.writerow(['filename', 'binding_energy_ddG'])
        for filename, ddg in results:
            writer.writerow([filename, f"{ddg:.4f}"])

    # Print summary
    print(f"\nResults summary:")
    print(f"  Output file: {args.output_csv}")
    print(f"  Files processed: {len(results)}")
    print(f"  Files skipped: {errors}")
    print(f"\nTop 3 strongest binders (most negative ddG):")
    for filename, ddg in results[:3]:
        print(f"  {filename}: {ddg:.4f}")

    return 0


if __name__ == "__main__":
    sys.exit(main())

    for i, data_line in enumerate(data_lines):
        values = re.split(r'\s+', data_line[6:].strip())
        if len(values) != len(headers):
            raise ValueError(f"Data line {i+1} in {basename} has {len(values)} "
                            f"columns, expected {len(headers)}.")

        try:
            ddg = float(values[ddg_idx])
        except (ValueError, IndexError):
            raise ValueError(f"Cannot parse ddG value in data line {i+1} "
                            f"of {basename}: '{values[ddg_idx]}'")

        if not isfinite(ddg):
            print(f"Warning: Non-finite ddG ({ddg}) found in data line "
                  f"{i+1} of {basename}, skipping this line.", file=sys.stderr)
            continue

        ddg_values.append(ddg)

    if not ddg_values:
        raise ValueError(f"No valid ddG values found in {basename}")

    # If multiple data lines, take the mean and warn
    if len(ddg_values) > 1:
        print(f"Note: {basename} contains {len(ddg_values)} data lines. "
              f"Using mean ddG = {mean(ddg_values):.4f}.", file=sys.stderr)

    # Use mean of all ddG values
    return basename, mean(ddg_values)


def main() -> int:
    """Main entry point. Returns 0 on success, 1 on failure."""
    parser = argparse.ArgumentParser(
        description=(
            "Process Rosetta .sc score files and output sorted CSV. "
            "Sorts from strongest to weakest binding "
            "(most negative to most positive ddG). "
            "Multiple data lines per file are averaged."
        )
    )
    parser.add_argument('input_dir', help="Directory containing .sc files")
    parser.add_argument('output_csv', help="Path to output CSV file")
    args = parser.parse_args()

    # Validate input directory
    if not os.path.isdir(args.input_dir):
        print(f"Error: Directory not found: {args.input_dir}", file=sys.stderr)
        return 1

    # Check if output file already exists
    if os.path.exists(args.output_csv):
        response = input(f"Warning: {args.output_csv} already exists. "
                        f"Overwrite? [y/N]: ").strip().lower()
        if response != 'y':
            print("Aborted.", file=sys.stderr)
            return 1

    # Find all .sc files
    score_files = sorted(glob.glob(os.path.join(args.input_dir, '*.sc')))
    if not score_files:
        print(f"Error: No .sc files found in {args.input_dir}", file=sys.stderr)
        return 1

    print(f"Found {len(score_files)} score files to process.")

    # Parse all score files
    results: List[Tuple[str, float]] = []
    errors = 0
    for filepath in score_files:
        try:
            results.append(parse_rosetta_score_file(filepath))
        except ValueError as e:
            print(f"Error: {e}", file=sys.stderr)
            errors += 1

    if not results:
        print("Error: No valid score files could be parsed.", file=sys.stderr)
        return 1

    # Sort from most negative to most positive (strongest to weakest binding)
    results.sort(key=lambda x: x[1])

    # Write CSV
    with open(args.output_csv, 'w', newline='') as f:
        writer = csv.writer(f)
        # Write metadata as a comment line (RFC 4180 allows this)
        writer.writerow([
            '# Sorted from strongest binding (most negative ddG) '
            'to weakest binding (most positive ddG)'
        ])
        writer.writerow(['filename', 'binding_energy_ddG'])
        for filename, ddg in results:
            writer.writerow([filename, f"{ddg:.4f}"])

    # Print summary
    print(f"\nResults summary:")
    print(f"  Output file: {args.output_csv}")
    print(f"  Files processed: {len(results)}")
    print(f"  Files skipped: {errors}")
    print(f"\nTop 3 strongest binders (most negative ddG):")
    for filename, ddg in results[:3]:
        print(f"  {filename}: {ddg:.4f}")

    return 0


if __name__ == "__main__":
    sys.exit(main())
