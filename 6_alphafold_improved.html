#!/usr/bin/env python3
"""
Calculate interface pLDDT from AlphaFold-Multimer predicted nanobody-antigen complexes
and generate ranked comparisons across multiple SARS-CoV-2 variants.

This script processes PDB files containing predicted nanobody-antigen complex structures
from AlphaFold-Multimer and computes the interface pLDDT score - the average pLDDT
confidence score for residues at the nanobody-antigen interface.

Workflow for Research Context:
    The team has modified existing nanobodies to improve binding to the newest
    SARS-CoV-2 variant while maintaining activity against other variants. This script
    helps evaluate these designs by:
    
    1. Processing AlphaFold-Multimer predictions for each nanobody-variant pair
    2. Calculating interface pLDDT as a confidence metric for each predicted complex
    3. Allowing cross-variant comparison to identify broadly effective nanobodies
    
    To use for this purpose:
    - Name PDB files consistently, e.g.:
        Nb1_BA.2.86.pdb, Nb1_JN.1.pdb, Nb1_XBB.1.5.pdb
        Nb2_BA.2.86.pdb, Nb2_JN.1.pdb, Nb2_XBB.1.5.pdb
    - The --variant-map option can be used to explicitly label variants
    - The script will produce a summary ranking nanobody designs by their
      average and minimum interface pLDDT across all tested variants

Scientific Background:
    Interface pLDDT is a prediction confidence metric, NOT a binding affinity measure.
    It quantifies how confident AlphaFold-Multimer is in its predicted structure at the
    binding interface. Higher scores (>80) indicate the model is confident in the
    relative positioning of the two chains, while lower scores (<60) suggest the
    predicted binding mode may be unreliable.

    Important Caveats:
    - Interface pLDDT correlates weakly with experimental binding affinity
    - Differences of <5-10 points between designs may not be significant
    - Multiple independent predictions (different seeds) improve reliability
    - Experimental validation is required for final candidate selection

    Interface Definition:
    We define interface residues as those with any heavy atom within a distance cutoff
    (default 5.0 Å) of any heavy atom in the other chain. This contact-based definition
    is appropriate for nanobody-antigen complexes because:
    1. Nanobody binding typically involves a small, well-defined paratope (CDR loops)
       contacting a complementary epitope on the antigen
    2. The 5Å cutoff captures both direct hydrogen bonds and van der Waals contacts
    3. Heavy atoms only (no hydrogens) because hydrogen positions are not reliably
       determined from X-ray structures or predicted by AlphaFold

Usage:
    # Basic usage - process all PDB files in a directory
    python calculate_interface_plddt.py ./alphafold_outputs/ results.csv

    # With variant mapping for cross-variant comparison
    python calculate_interface_plddt.py ./outputs/ results.csv \\
        --variant-map variants.csv

    # Specify chain IDs (useful if auto-detection gives warnings)
    python calculate_interface_plddt.py ./outputs/ results.csv \\
        --nanobody-chain A --antigen-chain B

    # Custom distance cutoff (e.g., 4.0 for tighter contact definition)
    python calculate_interface_plddt.py ./outputs/ results.csv \\
        --distance-cutoff 4.0

Variant Map CSV Format:
    filename,variant,nanobody_design
    Nb1_BA286.pdb,BA.2.86,Nb1
    Nb1_JN1.pdb,JN.1,Nb1
    Nb2_BA286.pdb,BA.2.86,Nb2
    Nb2_JN1.pdb,JN.1,Nb2

Options:
    --distance-cutoff FLOAT  Distance cutoff in Å for interface (default: 5.0)
    --nanobody-chain STR     Chain ID for nanobody (auto-detected if not provided)
    --antigen-chain STR      Chain ID for antigen (auto-detected if not provided)
    --variant-map PATH       CSV file mapping PDB filenames to variant/design info
"""

def find_pdb_files(input_dir: str) -> List[Path]:
    """
    Find all PDB files in the input directory.
    
    Args:
        input_dir: Path to directory containing PDB files
        
    Returns:
        Sorted list of Path objects for all PDB files found
        
    Raises:
        ValueError: If directory doesn't exist or contains no PDB files
    """
    pdb_path = Path(input_dir)
    if not pdb_path.exists():
        raise ValueError(f"Input directory does not exist: {input_dir}")
    if not pdb_path.is_dir():
        raise ValueError(f"Input path is not a directory: {input_dir}")
    
    # Find both .pdb and .pdb.gz files
    pdb_files = list(pdb_path.glob("*.pdb")) + list(pdb_path.glob("*.pdb.gz"))
    
    if not pdb_files:
        raise ValueError(f"No PDB files found in {input_dir}")
    
    return sorted(pdb_files)


def load_variant_map(variant_map_path: str) -> Dict[str, Dict[str, str]]:
    """
    Load variant mapping CSV file.
    
    Expected columns: filename, variant, nanobody_design
    
    Args:
        variant_map_path: Path to CSV file
        
    Returns:
        Dictionary mapping filename to {variant, nanobody_design}
        
    Raises:
        ValueError: If file cannot be parsed or missing required columns
    """
    variant_data = {}
    required_columns = {'filename', 'variant', 'nanobody_design'}
    
    try:
        with open(variant_map_path, 'r') as f:
            reader = csv.DictReader(f)
            
            # Validate columns
            if not reader.fieldnames:
                raise ValueError("Empty CSV file")
            
            missing_cols = required_columns - set(reader.fieldnames)
            if missing_cols:
                raise ValueError(
                    f"Missing required columns: {missing_cols}. "
                    f"Required: {required_columns}"
                )
            
            for row in reader:
                filename = row['filename'].strip()
                if not filename:
                    continue
                
                variant_data[filename] = {
                    'variant': row['variant'].strip(),
                    'nanobody_design': row['nanobody_design'].strip()
                }
        
        if not variant_data:
            raise ValueError("No valid data rows found in variant map")
        
        return variant_data
        
    except FileNotFoundError:
        raise ValueError(f"Variant map file not found: {variant_map_path}")
    except csv.Error as e:
        raise ValueError(f"Error parsing variant map CSV: {e}")


def make_residue_key(chain_id: str, residue) -> str:
    """
    Create a canonical residue identifier handling insertion codes.
    
    PDB residues can have insertion codes (e.g., residue 100A vs 100B).
    This function creates a unique string key that handles this.
    
    Args:
        chain_id: Chain identifier string
        residue: Biopython Residue object
        
    Returns:
        Canonical key like "A:100" or "A:100A" if insertion code present
    """
    res_id = residue.get_id()
    res_num = res_id[1]
    insertion_code = res_id[2]
    
    if insertion_code != " ":
        return f"{chain_id}:{res_num}{insertion_code}"
    else:
        return f"{chain_id}:{res_num}"


def extract_plddt_from_pdb(structure: Structure) -> Dict[str, float]:
    """
    Extract per-residue pLDDT scores from the B-factor column.
    
    AlphaFold-Multimer stores the predicted LDDT score (0-100) in the 
    B-factor column of all ATOM records. We use the CA atom's B-factor
    as the residue-level pLDDT.
    
    Args:
        structure: Biopython Structure object
        
    Returns:
        Dictionary mapping residue identifier (chain:residue_number) to pLDDT score
        
    Raises:
        ValueError: If no valid pLDDT scores are found or scores are out of range
    """
    plddt_scores = {}
    
    for model in structure:
        for chain in model:
            for residue in chain:
                # Skip heteroatoms (water, ligands)
                if residue.get_id()[0] != " ":
                    continue
                
                # Create canonical residue key with insertion code handling
                res_key = make_residue_key(chain.get_id(), residue)
                
                # Get pLDDT from CA atom B-factor (standard AlphaFold convention)
                try:
                    ca_atom = residue["CA"]
                    plddt = ca_atom.get_bfactor()
                except KeyError:
                    # Fall back to first atom if no CA (e.g., glycine)
                    atoms = list(residue.get_atoms())
                    if atoms:
                        plddt = atoms[0].get_bfactor()
                    else:
                        continue
                
                # Validate pLDDT is in expected range
                if not (0 <= plddt <= 100):
                    print(
                        f"Warning: Unexpected pLDDT value {plddt:.1f} at "
                        f"{res_key}",
                        file=sys.stderr
                    )
                
                plddt_scores[res_key] = plddt
    
    if not plddt_scores:
        raise ValueError("No pLDDT scores found in structure")
    
    return plddt_scores


def count_protein_residues(chain) -> int:
    """
    Count the number of standard amino acid residues in a chain.
    
    Args:
        chain: Biopython Chain object
        
    Returns:
        Number of protein residues (excluding heteroatoms)
    """
    return sum(1 for residue in chain if residue.get_id()[0] == " ")


def validate_chain_assignment(nanobody_chain, antigen_chain) -> Optional[str]:
    """
    Validate that chain assignment makes biological sense.
    
    Checks the size ratio of nanobody to antigen to detect potential
    chain swaps. A nanobody should be significantly smaller than the
    antigen (typical ratio < 0.5).
    
    Args:
        nanobody_chain: Putative nanobody chain
        antigen_chain: Putative antigen chain
        
    Returns:
        Warning message string, or None if validation passes
    """
    nb_size = count_protein_residues(nanobody_chain)
    ag_size = count_protein_residues(antigen_chain)
    
    # Check for potential chain swap: nanobody should be smaller
    size_ratio = nb_size / max(ag_size, 1)
    
    if size_ratio > 0.8:
        return (
            f"Chains have similar size (nanobody: {nb_size}, antigen: {ag_size}). "
            f"Chain assignment may be incorrect. Use --nanobody-chain and "
            f"--antigen-chain to specify explicitly."
        )
    
    if nb_size < NANOBODY_MIN_RESIDUES:
        return (
            f"Nanobody chain has only {nb_size} residues "
            f"(expected {NANOBODY_MIN_RESIDUES}-{NANOBODY_MAX_RESIDUES}). "
            f"This may be a fragment rather than a complete nanobody."
        )
    
    if nb_size > NANOBODY_MAX_RESIDUES:
        return (
            f"Nanobody chain has {nb_size} residues, "
            f"which is large for a nanobody (max {NANOBODY_MAX_RESIDUES}). "
            f"Check chain assignment."
        )
    
    return None


def identify_chains(
    structure: Structure,
    specified_nanobody: Optional[str] = None,
    specified_antigen: Optional[str] = None
) -> Tuple[str, str, Optional[str]]:
    """
    Identify nanobody and antigen chains with user override option.
    
    If chain IDs are specified, they are validated and returned.
    Otherwise, auto-detect by chain size (nanobody is smaller protein chain).
    
    Args:
        structure: Biopython Structure object
        specified_nanobody: Optional user-specified nanobody chain ID
        specified_antigen: Optional user-specified antigen chain ID
        
    Returns:
        Tuple of (nanobody_chain_id, antigen_chain_id, warning_message)
        
    Raises:
        ValueError: If chains cannot be identified or specified chains don't exist
    """
    model = next(structure.get_models())
    chains = list(model.get_chains())
    
    # Filter to only protein chains (ignore water, ligands)
    protein_chains = []
    for chain in chains:
        if count_protein_residues(chain) > 0:
            protein_chains.append(chain)
    
    if len(protein_chains) < 2:
        raise ValueError(
            f"Need at least 2 protein chains, found {len(protein_chains)} "
            f"in {len(chains)} total chains"
        )
    
    # Use user-specified chains if provided
    if specified_nanobody and specified_antigen:
        chain_ids = [c.get_id() for c in protein_chains]
        if specified_nanobody not in chain_ids:
            raise ValueError(
                f"Specified nanobody chain '{specified_nanobody}' not found. "
                f"Available chains: {chain_ids}"
            )
        if specified_antigen not in chain_ids:
            raise ValueError(
                f"Specified antigen chain '{specified_antigen}' not found. "
                f"Available chains: {chain_ids}"
            )
        
        nb_chain = model[specified_nanobody]
        ag_chain = model[specified_antigen]
        warning = validate_chain_assignment(nb_chain, ag_chain)
        return specified_nanobody, specified_antigen, warning
    
    # Auto-detect: assume nanobody is the smaller chain
    protein_chains.sort(key=lambda c: count_protein_residues(c))
    
    nb_chain = protein_chains[0]
    ag_chain = protein_chains[1]
    
    warning = validate_chain_assignment(nb_chain, ag_chain)
    
    # Warn about extra chains
    if len(protein_chains) > 2:
        extra_ids = [c.get_id() for c in protein_chains[2:]]
        extra_warning = (
            f"Found {len(protein_chains)} protein chains. "
            f"Using '{nb_chain.get_id()}' as nanobody and "
            f"'{ag_chain.get_id()}' as antigen. "
            f"Ignored chains: {extra_ids}. "
            f"Use --nanobody-chain and --antigen-chain to specify explicitly."
        )
        warning = f"{extra_warning} | {warning}" if warning else extra_warning
    
    return nb_chain.get_id(), ag_chain.get_id(), warning


def get_interface_residues(
    structure: Structure,
    nanobody_chain_id: str,
    antigen_chain_id: str,
    distance_cutoff: float
) -> Tuple[List[str], List[str], int]:
    """
    Identify interface residues using NeighborSearch for efficiency.
    
    Uses Biopython's NeighborSearch with KD-tree for efficient spatial queries.
    Interface residues are defined as those with any heavy atom within the
    distance cutoff of any heavy atom in the other chain.
    
    Args:
        structure: Biopython Structure object
        nanobody_chain_id: Chain ID for the nanobody
        antigen_chain_id: Chain ID for the antigen
        distance_cutoff: Distance cutoff in Angstroms
        
    Returns:
        Tuple of:
        - List of residue keys for nanobody interface residues
        - List of residue keys for antigen interface residues
        - Number of inter-chain atom contacts within cutoff
    """
    model = next(structure.get_models())
    nb_chain = model[nanobody_chain_id]
    ag_chain = model[antigen_chain_id]
    
    # Build atom lists, storing residue key as an attribute on each atom
    nb_atoms = []
    ag_atoms = []
    
    for residue in nb_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(nanobody_chain_id, residue)
        for atom in residue:
            if atom.element != "H":  # Heavy atoms only
                atom.xtra['residue_key'] = res_key
                nb_atoms.append(atom)
    
    for residue in ag_chain:
        if residue.get_id()[0] != " ":
            continue
        res_key = make_residue_key(antigen_chain_id, residue)
        for atom in residue:
            if atom.element != "H":
                atom.xtra['residue_key'] = res_key
                ag_atoms.append(atom)
    
    if not nb_atoms or not ag_atoms:
        return [], [], 0
    
    # Build NeighborSearch for antigen atoms
    neighbor_search = NeighborSearch(ag_atoms)
    
    # Find all nanobody atoms within distance cutoff of any antigen atom
    nb_interface_residues = set()
    ag_interface_residues = set()
    contact_count = 0
    
    for nb_atom in nb_atoms:
        close_atoms = neighbor_search.search(nb_atom.get_coord(), distance_cutoff)
        if close_atoms:
            nb_interface_residues.add(nb_atom.xtra['residue_key'])
            contact_count += len(close_atoms)
            for ag_atom in close_atoms:
                ag_interface_residues.add(ag_atom.xtra['residue_key'])
    
    return (
        list(nb_interface_residues),
        list(ag_interface_residues),
        contact_count
    )


def calculate_interface_plddt(
    pdb_file: Path,
    distance_cutoff: float,
    specified_nanobody: Optional[str] = None,
    specified_antigen: Optional[str] = None
) -> Optional[Dict[str, Any]]:
    """
    Calculate interface pLDDT for a single PDB file.
    
    Returns None if processing fails (error printed to stderr).
    Otherwise returns a dictionary with results.
    
    Args:
        pdb_file: Path to PDB file
        distance_cutoff: Interface distance cutoff in Å
        specified_nanobody: Optional user-specified nanobody chain
        specified_antigen: Optional user-specified antigen chain
        
    Returns:
        Dictionary with keys:
        - 'filename': Name of PDB file
        - 'interface_pLDDT': Average pLDDT of interface residues (0-100)
        - 'overall_pLDDT': Average pLDDT of all residues (0-100)
        - 'num_interface_residues': Total residues at interface
        - 'num_nanobody_interface': Number of nanobody interface residues
        - 'num_antigen_interface': Number of antigen interface residues
        - 'nanobody_chain': Chain ID used for nanobody
        - 'antigen_chain': Chain ID used for antigen
        - 'num_contacts': Number of inter-chain atom contacts
        - 'warning': Any warnings about chain identification
        
    Returns None on failure.
    """
    try:
        # Parse PDB file
        parser = PDB.PDBParser(QUIET=True)
        structure = parser.get_structure("complex", str(pdb_file))
        
        # Extract pLDDT scores (raises ValueError if invalid)
        plddt_scores = extract_plddt_from_pdb(structure)
        
        # Identify chains (raises ValueError if can't identify)
        nb_chain, ag_chain, warning = identify_chains(
            structure, specified_nanobody, specified_antigen
        )
        
        # Calculate overall pLDDT
        overall_plddt = float(np.mean(list(plddt_scores.values())))
        
        # Identify interface residues
        nb_interface, ag_interface, num_contacts = get_interface_residues(
            structure, nb_chain, ag_chain, distance_cutoff
        )
        
        all_interface_residues = nb_interface + ag_interface
        
        if not all_interface_residues:
            print(
                f"Warning: {pdb_file.name}: No interface residues found "
                f"with distance cutoff {distance_cutoff}Å",
                file=sys.stderr
            )
            return None
        
        # Calculate interface pLDDT
        interface_plddt_values = [
            plddt_scores[res_key] for res_key in all_interface_residues
            if res_key in plddt_scores
        ]
        
        if not interface_plddt_values:
            print(
                f"Warning: {pdb_file.name}: No pLDDT scores found for interface residues",
                file=sys.stderr
            )
            return None
        
        interface_plddt = float(np.mean(interface_plddt_values))
        
        return {
            'filename': pdb_file.name,
            'interface_pLDDT': round(interface_plddt, 2),
            'overall_pLDDT': round(overall_plddt, 2),
            'num_interface_residues': len(all_interface_residues),
            'num_nanobody_interface': len(nb_interface),
            'num_antigen_interface': len(ag_interface),
            'nanobody_chain': nb_chain,
            'antigen_chain': ag_chain,
            'num_contacts': num_contacts,
            'warning': warning if warning else ''
        }
        
    except ValueError as e:
        print(f"Error: {pdb_file.name}: {e}", file=sys.stderr)
        return None
    except Exception as e:
        print(f"Error: {pdb_file.name}: Unexpected error: {e}", file=sys.stderr)
        return None


def generate_cross_variant_summary(
    results: List[Dict[str, Any]],
    variant_map: Dict[str, Dict[str, str]]
) -> List[Dict[str, Any]]:
    """
    Generate a ranked summary of nanobody designs across variants.
    
    For each nanobody design, calculates:
    - Mean interface pLDDT across all tested variants
    - Minimum interface pLDDT (worst-case performance)
    - Standard deviation (consistency across variants)
    
    Designs are ranked by a composite score: mean - std (penalizes inconsistency)
    
    Args:
        results: List of per-file result dictionaries
        variant_map: Dictionary mapping filenames to variant metadata
        
    Returns:
        List of summary dictionaries sorted by composite score (descending)
    """
    # Group results by nanobody design
    design_scores = defaultdict(list)
    
    for result in results:
        filename = result['filename']
        if filename in variant_map:
            design = variant_map[filename]['nanobody_design']
            variant = variant_map[filename]['variant']
            design_scores[design].append({
                'variant': variant,
                'interface_pLDDT': result['interface_pLDDT'],
                'filename': filename
            })
    
    if not design_scores:
        return []
    
    # Calculate summary statistics per design
    summaries = []
    for design, scores in design_scores.items():
        plddt_values = [s['interface_pLDDT'] for s in scores]
        mean_plddt = np.mean(plddt_values)
        min_plddt = min(plddt_values)
        std_plddt = np.std(plddt_values)
        
        # Composite score: high mean with low variance is best
        composite_score = mean_plddt - std_plddt
        
        # List variants and their scores
        variant_details = "; ".join(
            f"{s['variant']}={s['interface_pLDDT']:.1f}"
            for s in sorted(scores, key=lambda x: x['variant'])
        )
        
        summaries.append({
            'nanobody_design': design,
            'mean_interface_pLDDT': round(mean_plddt, 2),
            'min_interface_pLDDT': round(min_plddt, 2),
            'std_interface_pLDDT': round(std_plddt, 2),
            'composite_score': round(composite_score, 2),
            'num_variants_tested': len(scores),
            'variant_details': variant_details
        })
    
    # Sort by composite score (descending)
    summaries.sort(key=lambda x: x['composite_score'], reverse=True)
    
    return summaries


def main():
    """Main function to process PDB files and generate CSV output."""
    args = parse_args()
    
    # Validate parameters
    if args.distance_cutoff <= 0:
        print("Error: Distance cutoff must be positive", file=sys.stderr)
        sys.exit(1)
    
    if args.distance_cutoff > 10:
        print(
            f"Warning: Distance cutoff of {args.distance_cutoff} Å is unusually large. "
            f"Typical values are 4-6 Å for residue-residue contacts.",
            file=sys.stderr
        )
    
    # Load variant map if provided
    variant_map = None
    if args.variant_map:
        try:
            variant_map = load_variant_map(args.variant_map)
            print(f"Loaded variant map with {len(variant_map)} entries")
        except ValueError as e:
            print(f"Error loading variant map: {e}", file=sys.stderr)
            sys.exit(1)
    
    # Find all PDB files
    try:
        pdb_files = find_pdb_files(args.input_dir)
    except ValueError as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
    
    print(f"Found {len(pdb_files)} PDB files to process...")
    print(f"Interface distance cutoff: {args.distance_cutoff} Å")
    if args.nanobody_chain:
        print(f"Nanobody chain: {args.nanobody_chain}")
    if args.antigen_chain:
        print(f"Antigen chain: {args.antigen_chain}")

def make_residue_key(chain_id: str, residue) -> str:

def extract_plddt_from_pdb(structure: Structure) -> Dict[str, float]:
    """
    Extract per-residue pLDDT scores from the B-factor column.
    
    AlphaFold-Multimer stores the predicted LDDT score (0-100) in the 
    B-factor column of all ATOM records. We use the CA atom's B-factor
    as the residue-level pLDDT.
    
    Args:
        structure: Biopython Structure object
        
    Returns:
        Dictionary mapping residue identifier (chain:residue_number) to pLDDT score
        
    Raises:
        ValueError: If no valid pLDDT scores are found or scores are out of range
    """
    plddt_scores = {}
    
    for model in structure:
        for chain in model:
            for residue in chain:
                # Skip heteroatoms (water, ligands)
                if residue.get_id()[0] != " ":
                    continue
                
                # Create canonical residue key with insertion code handling
                res_key = make_residue_key(chain.get_id(), residue)
                
                # Get pLDDT from CA atom B-factor (standard AlphaFold convention)
                try:
                    ca_atom = residue["CA"]
                    plddt = ca_atom.get_bfactor()
                except KeyError:
                    # Fall back to first atom if no CA (e.g., glycine)
                    atoms = list(residue.get_atoms())
                    if atoms:
                        plddt = atoms[0].get_bfactor()
                    else:
                        continue
                
                # Validate pLDDT is in expected range
                if not (0 <= plddt <= 100):
                    print(
                        f"Warning: Unexpected pLDDT value {plddt:.1f} at "
                        f"{res_key}",
                        file=sys.stderr
                    )
                
                plddt_scores[res_key] = plddt
    
    if not plddt_scores:
        raise ValueError("No pLDDT scores found in structure")
    
    return plddt_scores

def validate_chain_assignment(nanobody_chain, antigen_chain) -> Optional[str]:
    """
    Validate that chain assignment makes biological sense.
    
    Checks the size ratio of nanobody to antigen to detect potential
    chain swaps. A nanobody should be significantly smaller than the
    antigen (typical ratio < 0.5).

        nanobody_chain: Putative nanobody chain
        antigen_chain: Putative antigen chain
        
    Returns:
        Warning message string, or None if validation passes
    """
    nb_size = count_protein_residues(nanobody_chain)
    ag_size = count_protein_residues(antigen_chain)
    
    # Check for potential chain swap: nanobody should be smaller
    size_ratio = nb_size / max(ag_size, 1)
    
    if size_ratio > 0.8:
        return (
            f"Chains have similar size (nanobody: {nb_size}, antigen: {ag_size}). "
            f"Chain assignment may be incorrect. Use --nanobody-chain and "
            f"--antigen-chain to specify explicitly."

def get_interface_residues(

    antigen_chain_id: str,

    Uses Biopython's NeighborSearch with KD-tree for efficient spatial queries.
    Interface residues are defined as those with any heavy atom within the
    distance cutoff of any heavy atom in the other chain.

            continue

                atom.xtra['residue_key'] = res_key
                ag_atoms.append(atom)

        return [], [], 0, 0
    
    # Build NeighborSearch for antigen atoms
    neighbor_search = NeighborSearch(ag_atoms)

    nb_interface_residues = set()
    ag_interface_residues = set()
    nb_interface_atoms = set()
    ag_interface_atoms = set()
    contact_count = 0
    
    for nb_atom in nb_atoms:
        close_atoms = neighbor_search.search(nb_atom.get_coord(), distance_cutoff)
        if close_atoms:
            nb_interface_residues.add(nb_atom.xtra['residue_key'])
            nb_interface_atoms.add(nb_atom.get_full_id())
            contact_count += len(close_atoms)
            for ag_atom in close_atoms:
                ag_interface_residues.add(ag_atom.xtra['residue_key'])
                ag_interface_atoms.add(ag_atom.get_full_id())
    
    total_interface_atoms = len(nb_interface_atoms) + len(ag_interface_atoms)
    
    return (
        list(nb_interface_residues),
        list(ag_interface_residues),
        contact_count,
        total_interface_atoms
    )


def calculate_interface_plddt(
    pdb_file: Path,
    distance_cutoff: float,
    specified_nanobody: Optional[str] = None,
    specified_antigen: Optional[str] = None
) -> Optional[Dict[str, Any]]:
    """
    Calculate interface pLDDT for a single PDB file.
    
    Returns None if processing fails (error printed to stderr).
    Otherwise returns a dictionary with results.
    
    Args:
        pdb_file: Path to PDB file
        distance_cutoff: Interface distance cutoff in Å
        specified_nanobody: Optional user-specified nanobody chain
        specified_antigen: Optional user-specified antigen chain
        
    Returns:
        Dictionary with keys:
        - 'filename': Name of PDB file
        - 'interface_pLDDT': Average pLDDT of interface residues (0-100)
        - 'overall_pLDDT': Average pLDDT of all residues (0-100)
        - 'num_interface_residues': Total residues at interface
        - 'num_interface_atoms': Total atoms at interface
        - 'num_nanobody_interface': Number of nanobody interface residues
        - 'num_antigen_interface': Number of antigen interface residues
        - 'nanobody_chain': Chain ID used for nanobody
        - 'antigen_chain': Chain ID used for antigen
        - 'num_contacts': Number of inter-chain atom contacts
        - 'warning': Any warnings about chain identification
        
    Returns None on failure.
    """
    try:
        # Parse PDB file
        parser = PDB.PDBParser(QUIET=True)
        structure = parser.get_structure("complex", str(pdb_file))
        
        # Extract pLDDT scores (raises ValueError if invalid)
        plddt_scores = extract_plddt_from_pdb(structure)
        
        # Identify chains (raises ValueError if can't identify)
        nb_chain, ag_chain, warning = identify_chains(
            structure, specified_nanobody, specified_antigen
        )
        
        # Calculate overall pLDDT (average per atom, not per residue)
        all_plddt_values = list(plddt_scores.values())
        overall_plddt = float(np.mean(all_plddt_values))
        
        # Identify interface residues
        nb_interface, ag_interface, num_contacts, num_interface_atoms = get_interface_residues(
            structure, nb_chain, ag_chain, distance_cutoff
        )
        
        all_interface_residues = nb_interface + ag_interface
        
        if not all_interface_residues:
            print(
                f"Warning: {pdb_file.name}: No interface residues found "
                f"with distance cutoff {distance_cutoff}Å",
                file=sys.stderr
            )
            return None
        
        # Calculate interface pLDDT (average per atom)
        # We need to get pLDDT for each atom at the interface
        model = next(structure.get_models())
        nb_chain_obj = model[nb_chain]
        ag_chain_obj = model[ag_chain]
        
        interface_atom_plddt_values = []
        
        # Collect pLDDT values for nanobody interface atoms
        for residue in nb_chain_obj:
            if residue.get_id()[0] != " ":
                continue
            res_key = make_residue_key(nb_chain, residue)
            if res_key in nb_interface:
                for atom in residue:
                    if atom.element != "H":
                        interface_atom_plddt_values.append(atom.get_bfactor())
        
        # Collect pLDDT values for antigen interface atoms
        for residue in ag_chain_obj:
            if residue.get_id()[0] != " ":
                continue
            res_key = make_residue_key(ag_chain, residue)
            if res_key in ag_interface:
                for atom in residue:
                    if atom.element != "H":
                        interface_atom_plddt_values.append(atom.get_bfactor())
        
        if not interface_atom_plddt_values:
            print(
                f"Warning: {pdb_file.name}: No pLDDT scores found for interface atoms",
                file=sys.stderr
            )
            return None
        
        interface_plddt = float(np.mean(interface_atom_plddt_values))
        
        return {
            'filename': pdb_file.name,
            'interface_pLDDT': round(interface_plddt, 2),
            'overall_pLDDT': round(overall_plddt, 2),
            'num_interface_residues': len(all_interface_residues),
            'num_interface_atoms': num_interface_atoms,
            'num_nanobody_interface': len(nb_interface),
            'num_antigen_interface': len(ag_interface),
            'nanobody_chain': nb_chain,
            'antigen_chain': ag_chain,
            'num_contacts': num_contacts,
            'warning': warning if warning else ''
        }
        
    except ValueError as e:
        print(f"Error: {pdb_file.name}: {e}", file=sys.stderr)
        return None
    except Exception as e:
        print(f"Error: {pdb_file.name}: Unexpected error: {e}", file=sys.stderr)
        return None


def generate_cross_variant_summary(
    results: List[Dict[str, Any]],
    variant_map: Dict[str, Dict[str, str]]
) -> List[Dict[str, Any]]:
    """
    Generate a ranked summary of nanobody designs across variants.
    
    For each nanobody design, calculates:
    - Mean interface pLDDT across all tested variants
    - Minimum interface pLDDT (worst-case performance)
    - Standard deviation (consistency across variants)
    
    Designs are ranked by a composite score: mean - std (penalizes inconsistency)
    
    Args:
        results: List of per-file result dictionaries
        variant_map: Dictionary mapping filenames to variant metadata
        
    Returns:
        List of summary dictionaries sorted by composite score (descending)
    """
    # Group results by nanobody design
    design_scores = defaultdict(list)
    
    for result in results:
        filename = result['filename']
        if filename in variant_map:
            design = variant_map[filename]['nanobody_design']
            variant = variant_map[filename]['variant']
            design_scores[design].append({
                'variant': variant,
                'interface_pLDDT': result['interface_pLDDT'],
                'filename': filename
            })
    
    if not design_scores:
        return []
    
    # Calculate summary statistics per design
    summaries = []
    for design, scores in design_scores.items():
        plddt_values = [s['interface_pLDDT'] for s in scores]
        mean_plddt = np.mean(plddt_values)
        min_plddt = min(plddt_values)
        std_plddt = np.std(plddt_values)
        
        # Composite score: high mean with low variance is best
        composite_score = mean_plddt - std_plddt
        
        # List variants and their scores
        variant_details = "; ".join(
            f"{s['variant']}={s['interface_pLDDT']:.1f}"
            for s in sorted(scores, key=lambda x: x['variant'])
        )
        
        summaries.append({
            'nanobody_design': design,
            'mean_interface_pLDDT': round(mean_plddt, 2),
            'min_interface_pLDDT': round(min_plddt, 2),
            'std_interface_pLDDT': round(std_plddt, 2),
            'composite_score': round(composite_score, 2),
            'num_variants_tested': len(scores),
            'variant_details': variant_details
        })
    
    # Sort by composite score (descending)
    summaries.sort(key=lambda x: x['composite_score'], reverse=True)
    
    return summaries

    """Main function to process PDB files and generate CSV output."""
    args = parse_args()
    
    # Validate parameters
    if args.distance_cutoff <= 0:
        print("Error: Distance cutoff must be positive", file=sys.stderr)
        sys.exit(1)
    
    if args.distance_cutoff > 10:
        print(
            f"Warning: Distance cutoff of {args.distance_cutoff} Å is unusually large. "
            f"Typical values are 4-6 Å for residue-residue contacts.",
            file=sys.stderr
        )
    
    # Load variant map if provided
    variant_map = None
    if args.variant_map:
        try:
            variant_map = load_variant_map(args.variant_map)
            print(f"Loaded variant map with {len(variant_map)} entries")
        except ValueError as e:
            print(f"Error loading variant map: {e}", file=sys.stderr)
            sys.exit(1)
    
    # Find all PDB files
    try:
        pdb_files = find_pdb_files(args.input_dir)
    except ValueError as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)
    
    print(f"Found {len(pdb_files)} PDB files to process...")
    print(f"Interface distance cutoff: {args.distance_cutoff} Å")
    if args.nanobody_chain:
        print(f"Nanobody chain: {args.nanobody_chain}")
    if args.antigen_chain:
        print(f"Antigen chain: {args.antigen_chain}")
    print("-" * 60)
    
    # Process each PDB file sequentially to avoid "Too many open files" error
    results = []
    processed = 0
    failed = 0
    
    for pdb_file in pdb_files:
        result = calculate_interface_plddt(
            pdb_file,
            args.distance_cutoff,
            args.nanobody_chain,
            args.antigen_chain
        )
        
        if result is not None:
            results.append(result)
            processed += 1
            
            # Show variant info if available
            extra_info = ""
            if variant_map and result['filename'] in variant_map:
                info = variant_map[result['filename']]
                extra_info = f" [{info['nanobody_design']} vs {info['variant']}]"
            
            print(
                f"  {result['filename']}: interface pLDDT = {result['interface_pLDDT']:.1f} "
                f"({result['num_interface_residues']} interface residues, "
                f"{result['num_interface_atoms']} interface atoms){extra_info}",
                file=sys.stderr
            )
            if result['warning']:
                print(f"    Warning: {result['warning']}", file=sys.stderr)
        else:
            failed += 1
    
    print("-" * 60)
    
    if not results:
        print("No successful predictions to report.", file=sys.stderr)
        sys.exit(1)
    
    # Write per-file results to CSV
    fieldnames = [
        'filename',
        'interface_pLDDT',
        'overall_pLDDT',
        'num_interface_residues',
        'num_interface_atoms',
        'num_nanobody_interface',
        'num_antigen_interface',
        'num_contacts',
        'nanobody_chain',
        'antigen_chain',
        'warning'
    ]
    
    # Add variant columns if map was provided
    if variant_map:
        fieldnames.extend(['variant', 'nanobody_design'])
        for result in results:
            if result['filename'] in variant_map:
                info = variant_map[result['filename']]
                result['variant'] = info['variant']
