    <!-- 
    RosettaScripts XML for calculating binding energy of nanobody-antigen complex.
    
    This script performs:
    1. Preprocessing: FastRelax to remove structural clashes throughout the complex
    2. Interface analysis: Calculate binding energy (ddG) using InterfaceAnalyzer
    
    IMPORTANT ASSUMPTIONS:
    - The PDB file must contain exactly two protein chains: nanobody and antigen
    - The jump between these two chains is assumed to be jump 1 (the first
      inter-chain jump Rosetta detects). If your PDB has more than two chains
      or non-standard chain ordering, you must adjust the 'jump' parameter
      in the InterfaceAnalyzerMover below (e.g., jump=2 for the second interface).
    - Chain IDs must be single uppercase letters (A-Z)
    - Disulfide bonds should be detected via command-line flag: -detect_disulf true
    - More negative ddG = stronger binding
    
    Usage:
        rosetta_scripts.default.linuxgccrelease \
            -database $ROSETTA_DB \
            -parser:protocol this_file.xml \
            -in:file:s complex.pdb \
            -out:file:scorefile output.sc \
            -detect_disulf true
    -->

    <SCOREFXNS>
        <!-- Standard ref2015 score function - well-validated for binding energy calculations -->
        <ScoreFunction name="ref2015" weights="ref2015" />
    </SCOREFXNS>

    <TASKOPERATIONS>
        <!-- Prevent any design (we only want relaxation, not sequence changes) -->
        <RestrictToRepacking name="no_design" />
    </TASKOPERATIONS>

    <MOVERS>
        <!-- 
        Preprocessing: FastRelax with 5 repeats to resolve structural clashes.
        We relax the entire complex (no residue restrictions) because:
        - Input PDBs may have clashes anywhere, not just at the interface
        - This is simpler and more robust for screening purposes
        - Binding energy calculations are relative, so global relaxation is appropriate
        -->
        <FastRelax 
            name="relax"
            scorefxn="ref2015"
            repeats="5"
            task_operations="no_design"
        />

        InterfaceAnalyzerMover: Calculate binding energy (ddG).
        - jump=1: Assumes the nanobody and antigen are the only two chains.
          For multi-chain complexes, adjust this parameter.
        - pack_input: Repack side chains before analysis for fair comparison
        - pack_separated: Repack separated monomers to account for conformational changes
        - interface_sc: Compute interface score components for detailed analysis

Part 2: Rosetta Command 

#!/bin/bash
# 
# Usage: ./run_rosetta_binding.sh <input.pdb> [output.sc]
#
# Example:
#   export ROSETTA_BIN=/path/to/rosetta/main/source/bin/rosetta_scripts.default.linuxgccrelease
#   export ROSETTA_DB=/path/to/rosetta/main/database
#   export XML_SCRIPT=/path/to/calculate_binding_energy.xml
#   ./run_rosetta_binding.sh complex.pdb binding_scores.sc
#
# This script requires three environment variables to be set:
#   ROSETTA_BIN  - Path to the Rosetta scripts executable
#   ROSETTA_DB   - Path to the Rosetta database directory
#   XML_SCRIPT   - Path to the XML protocol file
#
# If output file is not specified, defaults to binding_scores.sc

# Fail on errors
set -euo pipefail

# Check required environment variables
: "${ROSETTA_BIN:?Error: ROSETTA_BIN environment variable not set}"
: "${ROSETTA_DB:?Error: ROSETTA_DB environment variable not set}"
: "${XML_SCRIPT:?Error: XML_SCRIPT environment variable not set}"

# Parse command-line arguments
INPUT_PDB="${1:?Error: No input PDB file specified}"
OUTPUT_SCORE="${2:-binding_scores.sc}"

# Validate input file
if [ ! -f "$INPUT_PDB" ]; then
    echo "Error: Input PDB file not found: $INPUT_PDB" >&2
    exit 1
fi

# Validate executable
if [ ! -x "$ROSETTA_BIN" ]; then
    echo "Error: Rosetta binary not executable: $ROSETTA_BIN" >&2
    exit 1
fi

# Validate database
if [ ! -d "$ROSETTA_DB" ]; then
    echo "Error: Rosetta database not found: $ROSETTA_DB" >&2
    exit 1
fi

# Validate XML script
if [ ! -f "$XML_SCRIPT" ]; then
    echo "Error: XML script not found: $XML_SCRIPT" >&2
    exit 1
fi

# Run Rosetta
"$ROSETTA_BIN" \
    -database "$ROSETTA_DB" \
    -parser:protocol "$XML_SCRIPT" \
    -in:file:s "$INPUT_PDB" \
    -out:file:scorefile "$OUTPUT_SCORE" \
    -nstruct 1 \
    -ex1 -ex2 \
    -use_input_sc \
    -ignore_unrecognized_res \
    -detect_disulf true \
    -overwrite

echo "Binding energy calculation complete. Results in: $OUTPUT_SCORE"

Part 3: Python Script for Processing Score Files 

#!/usr/bin/env python3
"""
Process Rosetta binding energy score files and output sorted results as CSV.

Reads all .sc files from a directory, extracts binding energy (ddG) values,
and outputs a CSV sorted from strongest to weakest binding
(most negative to most positive ddG).

Handles multiple data lines per file by taking the mean ddG value,
with a warning if multiple lines are found.

Usage:
    python process_rosetta_scores.py <input_directory> <output_csv>
"""

import argparse
import csv
import glob
import os
import re
import sys
from math import isfinite
from statistics import mean
from typing import List, Tuple


def parse_rosetta_score_file(filepath: str) -> Tuple[str, float]:
    """
    Parse a Rosetta score file to extract the binding energy (ddG).

    If multiple data lines are present (multiple decoys), returns the mean ddG.
    This handles cases where Rosetta generates multiple output models.

    Args:
        filepath: Path to Rosetta .sc score file

    Returns:
        Tuple of (basename, mean_ddG_value)

    Raises:
        ValueError: If parsing fails, ddG column not found, or all values are invalid
    """
    basename = os.path.basename(filepath)

    with open(filepath, 'r') as f:
        lines = f.readlines()

    if len(lines) < 2:
        raise ValueError(f"Insufficient lines ({len(lines)}) in {basename}. "
                        f"Expected at least a header and one data line.")

    # Find all SCORE lines (header and data lines)
    score_lines = []
    for line in lines:
        stripped = line.strip()
        if stripped.startswith('SCORE:'):
            score_lines.append(stripped)

    if len(score_lines) < 2:
        raise ValueError(f"Found {len(score_lines)} SCORE line(s) in {basename}. "
                        f"Need at least 2 (header + data).")

    # First SCORE line is the header, rest are data lines
    header_line = score_lines[0]
    data_lines = score_lines[1:]

    # Parse column headers
    headers = re.split(r'\s+', header_line[6:].strip())

    # Find ddG column
    try:
        ddg_idx = headers.index('ddG')
    except ValueError:
        raise ValueError(f"No 'ddG' column in {basename}. "
                        f"Available columns: {', '.join(headers)}")

    # Extract ddG values from all data lines
    ddg_values = []
    for i, data_line in enumerate(data_lines):
        values = re.split(r'\s+', data_line[6:].strip())
        if len(values) != len(headers):
            raise ValueError(f"Data line {i+1} in {basename} has {len(values)} "
                            f"columns, expected {len(headers)}.")

        try:
            ddg = float(values[ddg_idx])
        except (ValueError, IndexError):
            raise ValueError(f"Cannot parse ddG value in data line {i+1} "
                            f"of {basename}: '{values[ddg_idx]}'")

        if not isfinite(ddg):
            print(f"Warning: Non-finite ddG ({ddg}) found in data line "
                  f"{i+1} of {basename}, skipping this line.", file=sys.stderr)
            continue

        ddg_values.append(ddg)

    if not ddg_values:
        raise ValueError(f"No valid ddG values found in {basename}")

    # If multiple data lines, take the mean and warn
    if len(ddg_values) > 1:
        print(f"Note: {basename} contains {len(ddg_values)} data lines. "
              f"Using mean ddG = {mean(ddg_values):.4f}.", file=sys.stderr)

    # Use mean of all ddG values
    return basename, mean(ddg_values)


def main() -> int:
    """Main entry point. Returns 0 on success, 1 on failure."""
    parser = argparse.ArgumentParser(
        description=(
            "Process Rosetta .sc score files and output sorted CSV. "
            "Sorts from strongest to weakest binding "
            "(most negative to most positive ddG). "
            "Multiple data lines per file are averaged."
        )
    )
    parser.add_argument('input_dir', help="Directory containing .sc files")
    parser.add_argument('output_csv', help="Path to output CSV file")
    args = parser.parse_args()

    # Validate input directory
    if not os.path.isdir(args.input_dir):
        print(f"Error: Directory not found: {args.input_dir}", file=sys.stderr)
        return 1

    # Check if output file already exists
    if os.path.exists(args.output_csv):
        response = input(f"Warning: {args.output_csv} already exists. "
                        f"Overwrite? [y/N]: ").strip().lower()
        if response != 'y':
            print("Aborted.", file=sys.stderr)
            return 1

    # Find all .sc files
    score_files = sorted(glob.glob(os.path.join(args.input_dir, '*.sc')))
    if not score_files:
        print(f"Error: No .sc files found in {args.input_dir}", file=sys.stderr)
        return 1

    print(f"Found {len(score_files)} score files to process.")

    # Parse all score files
    results: List[Tuple[str, float]] = []
    errors = 0
    for filepath in score_files:
        try:
            results.append(parse_rosetta_score_file(filepath))
        except ValueError as e:
            print(f"Error: {e}", file=sys.stderr)
            errors += 1

    if not results:
        print("Error: No valid score files could be parsed.", file=sys.stderr)
        return 1

    # Sort from most negative to most positive (strongest to weakest binding)
    results.sort(key=lambda x: x[1])

    # Write CSV
    with open(args.output_csv, 'w', newline='') as f:
        writer = csv.writer(f)
        # Write metadata as a comment line (RFC 4180 allows this)
        writer.writerow([
            '# Sorted from strongest binding (most negative ddG) '
            'to weakest binding (most positive ddG)'
        ])
        writer.writerow(['filename', 'binding_energy_ddG'])
        for filename, ddg in results:
            writer.writerow([filename, f"{ddg:.4f}"])

    # Print summary
    print(f"\nResults summary:")
    print(f"  Output file: {args.output_csv}")
    print(f"  Files processed: {len(results)}")
    print(f"  Files skipped: {errors}")
    print(f"\nTop 3 strongest binders (most negative ddG):")
    for filename, ddg in results[:3]:
        print(f"  {filename}: {ddg:.4f}")

    return 0


if __name__ == "__main__":
    sys.exit(main())
