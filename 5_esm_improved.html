#!/usr/bin/env python3
"""
ESM-based Nanobody Mutation Predictor for SARS-CoV-2 Variant Targeting

This script uses ESM-2 to evaluate point mutations in nanobody sequences
specifically for improving binding to SARS-CoV-2 variants. It incorporates:
- Target variant specification (spike RBD mutations)
- Cross-reactivity analysis across multiple variants
- CDR region annotation for binding interface prioritization
- Evolutionarily plausible mutation identification
- Interface-aware scoring using cross-attention between nanobody and spike

SCIENTIFIC RATIONALE:
ESM log-likelihood ratios predict mutation tolerance (evolutionary plausibility).
For SARS-CoV-2 nanobody engineering:
1. Mutations in CDR regions are most likely to affect spike binding
2. Evolutionarily plausible mutations maintain stability while allowing
   functional exploration
3. Cross-reactivity across variants requires mutations compatible with
   multiple spike RBD sequences
4. Interface positions predicted via ESM cross-attention improve targeting

Usage:
    python esm_mutation_predictor.py --nanobody <sequence> --spike <spike_seq> --target_variant JN.1
    python esm_mutation_predictor.py --nanobody <sequence> --spike <spike_seq> \\

def get_amino_acid_group(aa: str) -> str:
    """Categorize amino acid by chemical properties."""
    for group, amino_acids in AMINO_ACID_GROUPS.items():
        if aa in amino_acids:
            return group
    return 'unknown'


def is_mutation_conservative(wt: str, mutant: str) -> bool:
    """Determine if a mutation is conservative (within same chemical group)."""
    return get_amino_acid_group(wt) == get_amino_acid_group(mutant)


def parse_cdr_regions(cdr_string: Optional[str], seq_len: int) -> List[Tuple[int, int]]:
    """Parse CDR region specifications from command-line argument.
    
    Uses raw sequence positions (1-indexed). Different numbering schemes
    (Kabat, Chothia, IMGT) may give different CDR assignments.
    
    Args:
        cdr_string: String like '26-32,49-55,99-108' or None
        seq_len: Length of the sequence for validation
        
    Returns:
        List of (start, end) tuples (0-indexed, inclusive)
    """
    if not cdr_string:
        return []
    
    regions = []
    for part in cdr_string.split(','):
        try:
            start, end = part.strip().split('-')
            start_idx = int(start) - 1
            end_idx = int(end) - 1
            if start_idx < 0 or end_idx >= seq_len or start_idx > end_idx:
                print(f"Warning: Invalid CDR region '{part}' for sequence length {seq_len}, skipping")
                continue
            regions.append((start_idx, end_idx))
        except (ValueError, IndexError):
            print(f"Warning: Invalid CDR region format '{part}', skipping")
    return regions


def is_in_cdr_region(position: int, cdr_regions: List[Tuple[int, int]]) -> bool:
    """Check if a position falls within any CDR region."""
    for start, end in cdr_regions:
        if start <= position <= end:
            return True
    return False


def get_contact_region_name(position: int) -> str:
    """Get the name of the contact region a position belongs to."""
    for region_name, (start, end) in NANOBODY_CONTACT_REGIONS.items():
        if start <= position + 1 <= end:
            return region_name
    return 'framework'


def compute_shannon_entropy(probabilities: np.ndarray) -> float:
    """Compute Shannon entropy from probability distribution."""
    probs = probabilities[probabilities > 1e-10]
    if len(probs) == 0:
        return 0.0
    return -np.sum(probs * np.log2(probs))


def sanitize_filename(name: str) -> str:
    """Sanitize a string for use as a filename."""
    return ''.join(c if c.isalnum() or c in '._-' else '_' for c in name)


def resolve_variant_name(name: str) -> str:
    """Resolve variant name or alias to canonical name."""
    name_lower = name.lower().replace('_', '').replace('.', '')
    if name_lower in VARIANT_ALIASES:
        return VARIANT_ALIASES[name_lower]
    # Try direct match
    for canonical in VARIANT_RBD_MUTATIONS:
        if name_lower == canonical.lower().replace('_', ''):
            return canonical
    raise ValueError(f"Unknown variant: '{name}'. Available: "
                     f"{', '.join(VARIANT_RBD_MUTATIONS.keys())}")


@dataclass
class MutationResult:
    """Container for a single mutation evaluation result."""
    position: int  # 0-indexed position in sequence
    wild_type: str
    mutant: str
    log_likelihood_ratio: float
    wild_type_log_likelihood: float
    wild_type_probability: float
    position_entropy: float
    in_cdr_region: bool
    contact_region: str  # CDR1, CDR2, CDR3, or framework
    mutation_type: str  # 'conservative' or 'non-conservative'
    is_interface: bool = False  # Predicted interface position
    cross_variant_score: float = 0.0  # Cross-reactivity score (higher = more cross-reactive)


class ESMMutationEvaluator:
    """
    Evaluates point mutations in nanobody sequences for SARS-CoV-2 targeting.
    
    Provides evolutionarily plausibility scores along with variant-specific
    context to help prioritize mutations for experimental validation.
    Uses ESM cross-attention to predict interface positions when spike sequence provided.
    """
    
    def __init__(self, model_name: str = "facebook/esm1v_t33_650M_UR90S_1", device: str = None):
        """
        Initialize ESM mutation evaluator.
        
        Args:
            model_name: ESM model name from HuggingFace
            device: Device to run model on
        """
        if device is None:
            self.device = "cuda" if torch.cuda.is_available() else "cpu"
        else:
            self.device = device
            
        print(f"Loading ESM model '{model_name}' on {self.device}...")
        self.tokenizer = EsmTokenizer.from_pretrained(model_name)
        self.model = EsmForMaskedLM.from_pretrained(model_name)
        self.model = self.model.to(self.device)
        self.model.eval()
        
        # Verify token mapping for standard amino acids
        for aa in AMINO_ACIDS:
            token_id = self.tokenizer.convert_tokens_to_ids(aa)
            if token_id is None or token_id == self.tokenizer.unk_token_id:
                raise ValueError(f"Amino acid '{aa}' not properly mapped in tokenizer vocabulary.")
        
        # Cache for evaluated sequences
        self._cache: Dict[str, List[MutationResult]] = {}
        
        print(f"Model loaded successfully on {self.device}.")
    
    def predict_interface_positions(
        self,
        nanobody_seq: str,
        spike_seq: str,
        top_k: int = 20
    ) -> List[int]:
        """Predict nanobody positions at the binding interface using ESM cross-attention.
        
        This method uses ESM's self-attention patterns to identify which nanobody
        residues have strong cross-attention to spike protein residues. We compute
        the average attention from each nanobody position to all spike positions
        across all attention heads and layers. Positions with highest cross-attention
        are predicted to be at the binding interface.
        
        Args:
            nanobody_seq: Nanobody amino acid sequence
            spike_seq: Spike protein amino acid sequence
            top_k: Number of interface positions to predict
            
        Returns:
            List of 0-indexed nanobody positions predicted to be at the interface
        """
        # Concatenate sequences
        combined_seq = nanobody_seq + spike_seq
        nanobody_len = len(nanobody_seq)
        
        # Tokenize the combined sequence
        inputs = self.tokenizer(combined_seq, return_tensors="pt", add_special_tokens=True).to(self.device)
        input_ids = inputs.input_ids
        
        # Run the model and get attention weights from the last layer
        with torch.no_grad():
            outputs = self.model(input_ids, output_attentions=True)
            # Get attention from last layer: (batch, heads, seq_len, seq_len)
            attention = outputs.attentions[-1]
        
        # Average over all attention heads
        avg_attention = attention.mean(dim=1).squeeze(0)  # (seq_len, seq_len)
        
        # Extract cross-attention: nanobody positions attending to spike positions
        # Token positions: [CLS] + nanobody_tokens + spike_tokens
        nb_start = 1
        nb_end = nb_start + nanobody_len
        spike_start = nb_end
        spike_end = avg_attention.shape[0] - 1  # Excluding [EOS] if present
        
        # For each nanobody position, compute average attention to all spike positions
        cross_attention_scores = []
        for nb_pos in range(nb_start, nb_end):
            attention_to_spike = avg_attention[nb_pos, spike_start:spike_end]
            mean_attention = attention_to_spike.mean().item()
            cross_attention_scores.append(mean_attention)
        
        # Select top_k positions with highest cross-attention
        if len(cross_attention_scores) < top_k:
            top_k = len(cross_attention_scores)
        
        top_indices = np.argsort(cross_attention_scores)[-top_k:][::-1]
        interface_positions = sorted(top_indices.tolist())
        
        print(f"Predicted {len(interface_positions)} interface positions (top {top_k} by cross-attention): "
              f"{[p+1 for p in interface_positions]}", file=sys.stderr)
        
        return interface_positions
    
    def compute_position_probabilities(self, sequence: str) -> Tuple[np.ndarray, np.ndarray]:
        """Compute probability distributions and log-likelihoods for each position.
        
        Uses batched masking for efficient computation.
        
        Args:
            sequence: Input amino acid sequence
            
        Returns:
            Tuple of (probability_matrix, log_likelihood_matrix) each of shape (seq_len, 20)
        """
        seq_len = len(sequence)
        
        # Tokenize the sequence
        tokenized = self.tokenizer(sequence, return_tensors="pt", add_special_tokens=True)
        input_ids = tokenized.input_ids[0].to(self.device)
        
        # Create batch of masked sequences (one per position)
        batch_input_ids = input_ids.unsqueeze(0).repeat(seq_len, 1)
        
        for i in range(seq_len):
            token_pos = i + 1  # +1 for <cls> token
            if token_pos < batch_input_ids.shape[1]:
                batch_input_ids[i, token_pos] = self.tokenizer.mask_token_id
        
        # Run model on batch
        with torch.no_grad():
            outputs = self.model(batch_input_ids)
            logits = outputs.logits  # (batch_size, seq_len, vocab_size)
        
        prob_matrix = np.zeros((seq_len, 20))
        ll_matrix = np.zeros((seq_len, 20))
        
        for i in range(seq_len):
            token_pos = i + 1
            logits_at_pos = logits[i, token_pos, :]
            log_probs = F.log_softmax(logits_at_pos, dim=-1)
            probs = torch.exp(log_probs)
            
            for j, aa in enumerate(AMINO_ACIDS):
                aa_token_id = self.tokenizer.convert_tokens_to_ids(aa)
                prob_matrix[i, j] = probs[aa_token_id].item()
                ll_matrix[i, j] = log_probs[aa_token_id].item()
        
        return prob_matrix, ll_matrix
    
    def compute_cross_variant_score(self, mutation: MutationResult,
                                   target_variant: str,
                                   cross_variants: List[str]) -> float:
        """
        Compute a simple cross-reactivity score based on position importance.
        
        Higher scores indicate mutations in regions that are conserved across
        variants, suggesting better cross-reactivity potential.
        
        Args:
            mutation: The mutation to evaluate
            target_variant: Primary target variant
            cross_variants: Additional variants for cross-reactivity
            
        Returns:
            Score from 0 to 1 (higher = better cross-reactivity)
        """
        # Framework regions are highly conserved across variants
        if mutation.contact_region == 'framework':
            return 1.0
        
        # CDR mutations get moderate scores (may need to adapt to different variants)
        if mutation.in_cdr_region:
            # Conservative mutations in CDRs are more likely cross-reactive
            if mutation.mutation_type == 'conservative':
                return 0.7
            return 0.4
        
        return 0.5
    
    def evaluate_mutations(self, sequence: str,
                          cdr_regions: Optional[List[Tuple[int, int]]] = None,
                          spike_seq: Optional[str] = None,
                          target_variant: str = 'JN_1',
                          cross_variants: Optional[List[str]] = None) -> List[MutationResult]:
        """
        Evaluate all possible point mutations with variant context.
        
        Args:
            sequence: Input nanobody sequence
            cdr_regions: CDR regions for annotation
            spike_seq: Spike protein sequence for interface prediction (optional)
            target_variant: Primary SARS-CoV-2 variant to target
            cross_variants: Additional variants for cross-reactivity analysis
            
        Returns:
            List of MutationResult objects, sorted by log-likelihood ratio
        """
        # Check cache
        cache_key = f"{sequence}_{target_variant}_{spike_seq[:50] if spike_seq else 'nospike'}"
        if cache_key in self._cache:
            return self._cache[cache_key]
        
        seq_len = len(sequence)
        sequence = sequence.upper()
        
        # Validate sequence
        valid_amino_acids = set(AMINO_ACIDS)
        for i, aa in enumerate(sequence):
            if aa not in valid_amino_acids:
                raise ValueError(f"Invalid amino acid '{aa}' at position {i}.")
        
        if cdr_regions is None:
            cdr_regions = []
        
        if cross_variants is None:
            cross_variants = []
        
        # Predict interface positions if spike sequence provided
        interface_positions = []
        if spike_seq:
            print("Predicting interface positions from cross-attention...")
            interface_positions = self.predict_interface_positions(sequence, spike_seq)
        
        # Compute probability distributions
        print("Computing position-wise probabilities with ESM...")
        prob_matrix, ll_matrix = self.compute_position_probabilities(sequence)
        
        # Get wild-type values
        wt_indices = [AMINO_ACIDS.index(aa) for aa in sequence]
        wt_probs = prob_matrix[np.arange(seq_len), wt_indices]
        wt_ll = ll_matrix[np.arange(seq_len), wt_indices]
        
        # Compute per-position entropy
        position_entropies = [compute_shannon_entropy(prob_matrix[i, :]) 
                            for i in range(seq_len)]
        
        mutations = []
        
        # Get variant information for context
        target_mutations = VARIANT_RBD_MUTATIONS.get(target_variant, [])
        
        print(f"Target variant: {target_variant} ({len(target_mutations)} RBD mutations)")
        if cross_variants:
            print(f"Cross-reactivity variants: {', '.join(cross_variants)}")
        
        # Evaluate all possible point mutations
        print("Evaluating all possible point mutations...")
        for pos in range(seq_len):
            wt_aa = sequence[pos]
            wt_ll_pos = wt_ll[pos]
            wt_prob_pos = wt_probs[pos]
            entropy_pos = position_entropies[pos]
            in_cdr = is_in_cdr_region(pos, cdr_regions)
            contact_region = get_contact_region_name(pos)
            is_interface = pos in interface_positions
            
            for j, mut_aa in enumerate(AMINO_ACIDS):
                if mut_aa == wt_aa:
                    continue
                
                mut_ll = ll_matrix[pos, j]
                ll_ratio = mut_ll - wt_ll_pos
                
                mutation = MutationResult(
                    position=pos,
                    wild_type=wt_aa,
                    mutant=mut_aa,
                    log_likelihood_ratio=ll_ratio,
                    wild_type_log_likelihood=wt_ll_pos,
                    wild_type_probability=wt_prob_pos,
                    position_entropy=entropy_pos,
                    in_cdr_region=in_cdr,
                    contact_region=contact_region,
                    mutation_type='conservative' if is_mutation_conservative(wt_aa, mut_aa) 
                                                  else 'non-conservative',
                    is_interface=is_interface
                )
                
                # Compute cross-reactivity score
                mutation.cross_variant_score = self.compute_cross_variant_score(
                    mutation, target_variant, cross_variants
                )
                
                mutations.append(mutation)
        
        # Sort by log-likelihood ratio descending
        mutations.sort(key=lambda x: x.log_likelihood_ratio, reverse=True)
        
        # Cache results
        self._cache[cache_key] = mutations
        
        return mutations
    
    def get_top_mutations(self, sequence: str, top_k: int = 10,
                         cdr_regions: Optional[List[Tuple[int, int]]] = None,
                         spike_seq: Optional[str] = None,
                         target_variant: str = 'JN_1',
                         cross_variants: Optional[List[str]] = None,
                         prioritize_cdr: bool = True,
                         prioritize_interface: bool = True) -> List[MutationResult]:
        """
        Get top K mutations with optional CDR and interface prioritization.
        
        Args:
            sequence: Input nanobody sequence
            top_k: Number of top mutations to return
            cdr_regions: CDR regions for annotation
            spike_seq: Spike protein sequence for interface prediction
            target_variant: Primary SARS-CoV-2 variant
            cross_variants: Additional variants for cross-reactivity
            prioritize_cdr: If True, list CDR mutations first (within top K)
            prioritize_interface: If True, prioritize interface mutations
            
        Returns:
            List of top MutationResult objects
        """
        all_mutations = self.evaluate_mutations(
            sequence, cdr_regions, spike_seq, target_variant, cross_variants
        )
        
        if prioritize_cdr and cdr_regions:
            # Separate CDR and non-CDR mutations
            cdr_mutations = [m for m in all_mutations if m.in_cdr_region]
            framework_mutations = [m for m in all_mutations if not m.in_cdr_region]
            
            # If interface prioritization, further separate interface CDR mutations
            if prioritize_interface and spike_seq:
                interface_cdr = [m for m in cdr_mutations if m.is_interface]
                non_interface_cdr = [m for m in cdr_mutations if not m.is_interface]
                
                # Take top K: interface CDR first, then other CDR, then framework
                result = interface_cdr[:top_k]
                if len(result) < top_k:
                    remaining = top_k - len(result)
                    result.extend(non_interface_cdr[:remaining])
                if len(result) < top_k:
                    remaining = top_k - len(result)
                    result.extend(framework_mutations[:remaining])
                return result
            
            # Take top K from CDR first, then fill with framework
            result = cdr_mutations[:top_k]
            if len(result) < top_k:
                remaining = top_k - len(result)
                result.extend(framework_mutations[:remaining])
            return result
        
        return all_mutations[:top_k]


def format_sequence(sequence: str) -> str:
    """Clean and validate input sequence."""
    sequence = sequence.strip().upper()
    sequence = ''.join(c for c in sequence if c.isalpha())
    return sequence

            cross_variants: Additional variants for cross-reactivity analysis
            
        Returns:
            List of MutationResult objects, sorted by log-likelihood ratio
        """
        # Check cache
        cache_key = f"{sequence}_{target_variant}_{spike_seq[:50] if spike_seq else 'nospike'}"
        if cache_key in self._cache:
            return self._cache[cache_key]
        
        seq_len = len(sequence)
        sequence = sequence.upper()
        
        # Validate sequence
        valid_amino_acids = set(AMINO_ACIDS)
        for i, aa in enumerate(sequence):
            if aa not in valid_amino_acids:
                raise ValueError(f"Invalid amino acid '{aa}' at position {i}.")
        
        if cdr_regions is None:
            cdr_regions = []
        
        if cross_variants is None:
            cross_variants = []
        
        # Predict interface positions if spike sequence provided
        interface_positions = []
        if spike_seq:
            print("Predicting interface positions from cross-attention...")
            interface_positions = self.predict_interface_positions(sequence, spike_seq)
        
        # Compute probability distributions
        print("Computing position-wise probabilities with ESM...")
        prob_matrix, ll_matrix = self.compute_position_probabilities(sequence)
        
        # Get wild-type values
        wt_indices = [AMINO_ACIDS.index(aa) for aa in sequence]
        wt_probs = prob_matrix[np.arange(seq_len), wt_indices]
        wt_ll = ll_matrix[np.arange(seq_len), wt_indices]
        
        # Compute per-position entropy
        position_entropies = [compute_shannon_entropy(prob_matrix[i, :]) 
                            for i in range(seq_len)]
        
        mutations = []
        
        # Get variant information for context
        target_mutations = VARIANT_RBD_MUTATIONS.get(target_variant, [])
        
        print(f"Target variant: {target_variant} ({len(target_mutations)} RBD mutations)")
        if cross_variants:
            print(f"Cross-reactivity variants: {', '.join(cross_variants)}")
        
        # Evaluate all possible point mutations
        print("Evaluating all possible point mutations...")
        for pos in range(seq_len):
            wt_aa = sequence[pos]
            wt_ll_pos = wt_ll[pos]
            wt_prob_pos = wt_probs[pos]
            entropy_pos = position_entropies[pos]
            in_cdr = is_in_cdr_region(pos, cdr_regions)
            contact_region = get_contact_region_name(pos)
            is_interface = pos in interface_positions
            
            for j, mut_aa in enumerate(AMINO_ACIDS):
                if mut_aa == wt_aa:
                    continue
                
                mut_ll = ll_matrix[pos, j]
                ll_ratio = mut_ll - wt_ll_pos
                
                mutation = MutationResult(
                    position=pos,
                    wild_type=wt_aa,
                    mutant=mut_aa,
                    log_likelihood_ratio=ll_ratio,
                    wild_type_log_likelihood=wt_ll_pos,
                    wild_type_probability=wt_prob_pos,
                    position_entropy=entropy_pos,
                    in_cdr_region=in_cdr,
                    contact_region=contact_region,
                    mutation_type='conservative' if is_mutation_conservative(wt_aa, mut_aa) 
                                                  else 'non-conservative',
                    is_interface=is_interface
                )
                
                # Compute cross-reactivity score
                mutation.cross_variant_score = self.compute_cross_variant_score(
                    mutation, target_variant, cross_variants
                )
                
                mutations.append(mutation)
        
        # Sort by log-likelihood ratio descending
        mutations.sort(key=lambda x: x.log_likelihood_ratio, reverse=True)
        
        # Cache results
        self._cache[cache_key] = mutations
        
        return mutations
    
    def get_top_mutations(self, sequence: str, top_k: int = 10,
                         cdr_regions: Optional[List[Tuple[int, int]]] = None,
                         spike_seq: Optional[str] = None,
                         target_variant: str = 'JN_1',
                         cross_variants: Optional[List[str]] = None,
                         prioritize_cdr: bool = True,
                         prioritize_interface: bool = True) -> List[MutationResult]:
        """
        Get top K mutations with optional CDR and interface prioritization.
        
        Args:
            sequence: Input nanobody sequence
            top_k: Number of top mutations to return
            cdr_regions: CDR regions for annotation
            spike_seq: Spike protein sequence for interface prediction
            target_variant: Primary SARS-CoV-2 variant
            cross_variants: Additional variants for cross-reactivity
            prioritize_cdr: If True, list CDR mutations first (within top K)
            prioritize_interface: If True, prioritize interface mutations
            
        Returns:
            List of top MutationResult objects
        """
        all_mutations = self.evaluate_mutations(
            sequence, cdr_regions, spike_seq, target_variant, cross_variants
        )
        
        if prioritize_cdr and cdr_regions:
            # Separate CDR and non-CDR mutations
            cdr_mutations = [m for m in all_mutations if m.in_cdr_region]
            framework_mutations = [m for m in all_mutations if not m.in_cdr_region]
            
            # If interface prioritization, further separate interface CDR mutations
            if prioritize_interface and spike_seq:
                interface_cdr = [m for m in cdr_mutations if m.is_interface]
                non_interface_cdr = [m for m in cdr_mutations if not m.is_interface]
                
                # Take top K: interface CDR first, then other CDR, then framework
                result = interface_cdr[:top_k]
                if len(result) < top_k:
                    remaining = top_k - len(result)
                    result.extend(non_interface_cdr[:remaining])
                if len(result) < top_k:
                    remaining = top_k - len(result)
                    result.extend(framework_mutations[:remaining])
                return result
            
            # Take top K from CDR first, then fill with framework
            result = cdr_mutations[:top_k]
            if len(result) < top_k:
                remaining = top_k - len(result)
                result.extend(framework_mutations[:remaining])
            return result
        
        return all_mutations[:top_k]


def format_sequence(sequence: str) -> str:
    """Clean and validate input sequence."""
    sequence = sequence.strip().upper()
    sequence = ''.join(c for c in sequence if c.isalpha())
    return sequence


def load_nanobody_csv(csv_path: str) -> List[Tuple[str, str]]:
    """Load nanobody sequences from a CSV file.
    
    Args:
        csv_path: Path to CSV file with columns 'sequence' and 'name'
        
    Returns:
        List of (name, sequence) tuples
    """
    sequences = []
    with open(csv_path, 'r') as f:
        reader = csv.DictReader(f)
        if 'sequence' not in reader.fieldnames or 'name' not in reader.fieldnames:
            raise ValueError("CSV file must contain 'sequence' and 'name' columns")
        for row in reader:
            name = row['name'].strip()
            sequence = format_sequence(row['sequence'])
            if len(sequence) < 10:
                print(f"Warning: Sequence '{name}' is too short (minimum 10 amino acids), skipping")
                continue
            sequences.append((name, sequence))
    return sequences


def save_mutation_results(mutations: List[MutationResult], 
                         nanobody_name: str, 
                         save_dir: str):
    """Save mutation results to a CSV file.
    
    Args:
        mutations: List of MutationResult objects
        nanobody_name: Name of the nanobody
        save_dir: Directory to save the CSV file
    """
    os.makedirs(save_dir, exist_ok=True)
    safe_name = sanitize_filename(nanobody_name)
    output_file = os.path.join(save_dir, f"{safe_name}.csv")
    
    with open(output_file, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['mutated_sequence', 'position', 'original_aa', 'mutated_aa', 'log_likelihood_ratio'])
        
        for mutation in mutations:
            # Create the mutated sequence representation
            mutated_seq = f"{mutation.wild_type}{mutation.position + 1}{mutation.mutant}"
            writer.writerow([
                mutated_seq,
                mutation.position + 1,  # 1-indexed position
                mutation.wild_type,
                mutation.mutant,
                f"{mutation.log_likelihood_ratio:.4f}"
            ])
    
    print(f"Mutation results saved to: {output_file}")
